    ```typescript
    export function buildServerWithHub(services: GridServices): { app: FastifyInstance; wsHub: WsHub } {
        const app = Fastify({ logger: false });
        const startedAt = Date.now();

        // ── EXISTING REST routes (keep ALL existing routes unchanged) ────────
        // … copy the existing route bodies here verbatim …

        // ── NEW: WebSocket endpoint /ws/events ───────────────────────────────
        // Register plugin (idempotent — only one call).
        void app.register(fastifyWebsocket, {
            options: { maxPayload: 1_048_576 },
        });

        // Construct hub AFTER register so the plugin is available to the route.
        const wsHub = new WsHub({ audit: services.audit, gridName: services.gridName });

        // Optional auth on upgrade (local-dev posture per 02-CONTEXT.md):
        //   - If GRID_WS_SECRET env is set: require Authorization: Bearer <secret>
        //     header OR ?token=<secret> query param. Missing/mismatch → 401.
        //   - If unset: permissive (developer default, 127.0.0.1 bind).
        app.register(async (instance) => {
            instance.get('/ws/events', { websocket: true }, (socket, req) => {
                const secret = process.env.GRID_WS_SECRET;
                if (secret) {
                    const headerAuth = (req.headers['authorization'] as string | undefined) ?? '';
                    const bearer = headerAuth.startsWith('Bearer ')
                        ? headerAuth.substring('Bearer '.length)
                        : null;
                    const queryToken =
                        typeof (req.query as any)?.token === 'string'
                            ? (req.query as any).token as string
                            : null;
                    const presented = bearer ?? queryToken;
                    if (presented !== secret) {
                        try { socket.close(1008, 'unauthorized'); } catch { /* swallow */ }
                        return;
                    }
                }
                // Adapter: fastify/ws socket → ServerSocket shape expected by WsHub.
                const adapter = {
                    get bufferedAmount() { return socket.bufferedAmount; },
                    send: (data: string) => socket.send(data),
                    close: (code?: number, reason?: string) => socket.close(code, reason),
                    on: (event: 'message' | 'close' | 'error', cb: any) => {
                        if (event === 'message') {
                            socket.on('message', (data: Buffer) => cb(data.toString('utf8')));
                        } else if (event === 'close') {
                            socket.on('close', () => cb());
                        } else {
                            socket.on('error', (err: Error) => cb(err));
                        }
                    },
                };
                wsHub.onConnect(adapter, { headers: req.headers as Record<string, unknown> });
            });
        });

        // ── Lifecycle: drain hub on app.close() ──────────────────────────────
        app.addHook('onClose', async () => {
            await wsHub.close();
        });

        return { app, wsHub };
    }
    ```

    ```typescript
    import { buildServerWithHub } from '../src/api/server.js';

    import { SpatialMap } from '../src/space/map.js';

    import { WebSocket as WsClient } from 'ws';
    import type { FastifyInstance } from 'fastify';

    function waitOpen(ws: WsClient, timeoutMs = 500): Promise<void> {
        return new Promise((resolve, reject) => {
            const t = setTimeout(() => reject(new Error('open timeout')), timeoutMs);
            ws.once('open', () => { clearTimeout(t); resolve(); });
            ws.once('error', (e) => { clearTimeout(t); reject(e); });
        });
    }
    function waitMessage(ws: WsClient, timeoutMs = 500): Promise<any> {
        return new Promise((resolve, reject) => {
            const t = setTimeout(() => reject(new Error('message timeout')), timeoutMs);
            ws.once('message', (buf) => { clearTimeout(t); resolve(JSON.parse(buf.toString())); });
            ws.once('error', (e) => { clearTimeout(t); reject(e); });
        });
    }
    function waitClose(ws: WsClient, timeoutMs = 1000): Promise<{ code: number; reason: string }> {
        return new Promise((resolve, reject) => {
            const t = setTimeout(() => reject(new Error('close timeout')), timeoutMs);
            ws.once('close', (code, reason) => {
                clearTimeout(t); resolve({ code, reason: reason.toString() });
            });
        });
    }

    describe('WS integration', () => {
        let app: FastifyInstance;
        let wsHub: WsHub;
        let audit: AuditChain;

            expect(hello.gridName).toBe('itest-grid');
            expect(typeof hello.lastEntryId).toBe('number');
            ws.close();
            await new Promise((r) => setTimeout(r, 50));
        });
    ```
