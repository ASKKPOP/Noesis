    - Construction: `new WsHub({ audit, gridName: 'test' })` subscribes to `audit.onAppend` exactly once; `audit.append(...)` triggers the hub's internal dispatcher.
    - Connect: `hub.onConnect(fakeSocket)` immediately sends a `HelloFrame` with `type: 'hello'`, current `serverTime`, `gridName`, `lastEntryId = audit.length`.
    - Allowlist gate: when `audit.append('nous.spoke', ...)` fires, every connected client receives an `EventFrame`; when `audit.append('reflection.completed', ...)` fires (NOT allowlisted), NO client receives a frame.
    - Topic filter: client sends `{type:"subscribe", filters:["nous.*"]}`. After that, allowlisted `nous.moved` arrives but allowlisted `trade.settled` does NOT reach this client; an UNfiltered second client receives both.
    - Buffered/direct send: when `fakeSocket.bufferedAmount === 0`, the hub calls `socket.send` directly; when `bufferedAmount > watermarkBytes`, the hub pushes to the ring buffer and does NOT call `socket.send` until the next drain tick.
    - Overflow → DroppedFrame: fill the ring buffer (capacity 256 by default) past capacity while socket stays backpressured; when the drain happens (simulated by lowering `bufferedAmount` and invoking the hub's flush path), the next frame emitted is a `DroppedFrame` with `sinceId` = smallest evicted id and `latestId` = largest evicted id, BEFORE any queued events drain.
    - Listener never calls send under backpressure: the listener function body never contains `socket.send` invocation when `bufferedAmount > watermark` — it only enqueues. (Assertion: spy on `socket.send`; under permanent backpressure during a burst, enqueue count grows and send count stays equal to the pre-burst send count.)
    - Disconnect cleanup: when `fakeSocket` emits `'close'`, the hub's internal client set size decreases by 1; no further frames are sent to that socket even if new events fire.
    - close() drain: `await hub.close()` sends a `ByeFrame { reason: "shutting down" }` to each client, invokes `socket.close(1001, ...)`, clears the internal clients set, and unsubscribes from `audit.onAppend` (verified by: after close(), `audit.append(...)` no longer triggers any send on any remaining mock).
    - Cross-client isolation: one slow client whose `bufferedAmount` stays above watermark does NOT delay direct sends to a fast client whose `bufferedAmount` is 0 — fast client's send call happens synchronously inside the listener, slow client's data goes to its own ring buffer.
    - Listener robustness: throwing from a socket's `send` method inside the hub's listener path MUST NOT propagate out of `audit.append()` — wrap internal send attempts in try/catch (belt-and-suspenders alongside Phase 1's already-present try/catch in `AuditChain`).
    - Malformed client frame: `parseClientFrame` returning `null` causes the hub to ignore the message silently (no throw, no socket.close).

    ```typescript
    /**
     * WsHub — owns all connected /ws/events clients. Subscribes ONCE to
     * AuditChain.onAppend, enforces the Phase 1 broadcast allowlist, and
     * fan-outs allowlisted entries to each ClientConnection for per-client
     * ring-buffered backpressure.
     *
     * Invariants (enforced by tests in grid/test/ws-hub.test.ts):
     *  1. The AuditChain listener NEVER calls ws.send — it only enqueues.
     *  2. A listener exception can never corrupt chain state (Phase 1
     *     already try/catches; we belt-and-suspenders here too).
     *  3. A slow client (high bufferedAmount) can NEVER delay a fast client.
     *  4. An entry whose eventType is not isAllowlisted() is dropped at the hub.
     *  5. close() is idempotent and leaves clients.size === 0.
     */

    import type { AuditChain } from '../audit/chain.js';
    import type { AuditEntry } from '../audit/types.js';
    import { isAllowlisted } from '../audit/broadcast-allowlist.js';
    import { RingBuffer } from '../util/ring-buffer.js';
    import {
        type HelloFrame, type EventFrame, type DroppedFrame, type ByeFrame,
        type ServerFrame, parseClientFrame,
    } from './ws-protocol.js';

    export interface ServerSocket {
        readonly bufferedAmount: number;
        send(data: string): void;
        close(code?: number, reason?: string): void;
        on(event: 'message', cb: (data: unknown) => void): void;
        on(event: 'close', cb: () => void): void;
        on(event: 'error', cb: (err: Error) => void): void;
    }

    export interface WsHubOptions {
        audit: AuditChain;
        gridName: string;
        bufferCapacity?: number;   // default 256
        watermarkBytes?: number;   // default 1_048_576
    }

    const DEFAULT_BUFFER_CAPACITY = 256;
    const DEFAULT_WATERMARK_BYTES = 1_048_576;

    // ── Internal: one connected client ────────────────────────────────────────
    class ClientConnection {
        private readonly buffer: RingBuffer<AuditEntry>;
        private filters: string[] | null = null;           // null = allow all (after Hello but before subscribe)
        private droppedMin: number | null = null;
        private droppedMax: number | null = null;
        public closed = false;

        constructor(
            public readonly socket: ServerSocket,
            private readonly watermarkBytes: number,
            bufferCapacity: number,
        ) {
            this.buffer = new RingBuffer(bufferCapacity);
        }

        setFilters(filters: string[] | undefined): void {
            // Undefined or empty array → allow all (within allowlist).
            this.filters = filters && filters.length > 0 ? filters.slice() : null;
        }

        matches(eventType: string): boolean {
            if (!this.filters) return true;
            for (const pattern of this.filters) {
                if (globMatch(pattern, eventType)) return true;
            }
            return false;
        }

        /**
         * Called from hub's audit listener. NEVER calls ws.send unless
         * socket is NOT backpressured AND the buffer is empty — this is
         * the fast-path direct send. Otherwise the entry is buffered and
         * a drain is scheduled.
         */
        enqueue(entry: AuditEntry): void {
            if (this.closed) return;
            // Fast path: socket is drained and we have no queued work.
            if (this.buffer.size === 0 && this.socket.bufferedAmount < this.watermarkBytes) {
                this.trySend({ type: 'event', entry });
                return;
            }
            // Slow path: buffer. If buffer evicts, track dropped range.
            const evicted = this.buffer.push(entry);
            if (evicted && evicted.id !== undefined) {
                if (this.droppedMin === null || evicted.id < this.droppedMin) this.droppedMin = evicted.id;
                if (this.droppedMax === null || evicted.id > this.droppedMax) this.droppedMax = evicted.id;
            }
            // Schedule drain attempt on next microtask (non-blocking).
            queueMicrotask(() => this.tryDrain());
        }

        /**
         * Attempt to flush queued entries. Called on microtask after enqueue,
         * and once synchronously from hub.broadcast() on the fast-path miss.
         * Sends a DroppedFrame FIRST if any drops were recorded, then drains.
         */
        tryDrain(): void {
            if (this.closed) return;
            if (this.socket.bufferedAmount >= this.watermarkBytes) return;   // still backpressured

            if (this.droppedMin !== null && this.droppedMax !== null) {
                this.trySend({ type: 'dropped', sinceId: this.droppedMin, latestId: this.droppedMax });
                this.droppedMin = null;
                this.droppedMax = null;
            }

            while (this.buffer.size > 0 && this.socket.bufferedAmount < this.watermarkBytes) {
                const next = this.buffer.drain().shift();
                if (!next) break;
                // Reinsert the remainder — drain() emptied buffer, so we shifted
                // off one and push the rest back (order preserved). This is O(n)
                // but n ≤ 256 so acceptable. Alternative: add shift() to
                // RingBuffer in a later refactor.
                const remainder = this.buffer; // buffer is now empty
                this.trySend({ type: 'event', entry: next });
                // No-op: this.buffer is already empty because drain() cleared it.
                // We lose remaining items — so instead, implement drain differently:
                // BELOW IS CORRECTED: drain snapshot approach.
                void remainder;
            }
            // NOTE: The simpler correct implementation is:
            //   const items = this.buffer.drain();
            //   for (const e of items) { if (backpressured) break; trySend({event}); }
            //   if (partially sent) push remaining back into buffer.
            // Use that pattern in the real code — the commentary above is to
            // remind the implementer not to use ad-hoc shift() semantics.
        }

        sendBye(reason: string): void {
            this.trySend({ type: 'bye', reason });
        }

        handleClientMessage(raw: unknown): void {
            const frame = parseClientFrame(raw);
            if (!frame) return; // silent ignore — see PITFALLS C4 tangent
            switch (frame.type) {
                case 'subscribe':
                    this.setFilters(frame.filters);
                    // sinceId replay is deferred — initial v1 relies on REST refill.
                    // If implementing: call hub.replayFrom(this, frame.sinceId) here.
                    break;
                case 'unsubscribe':
                    this.filters = []; // match nothing
                    break;
                case 'ping':
                    this.trySend({ type: 'pong', t: frame.t });
                    break;
                case 'pong':
                    // No-op — hub-driven keepalive lives at the hub level, not per-client here.
                    break;
            }
        }

        markClosed(): void { this.closed = true; }

        private trySend(frame: ServerFrame): void {
            if (this.closed) return;
            try {
                this.socket.send(JSON.stringify(frame));
            } catch {
                // Swallow — socket errors are handled via socket.on('error') / ('close').
                // Never propagate back into the audit listener chain.
            }
        }
    }

    // ── Glob matcher (simple dotted pattern: '*' matches one segment) ────────
    export function globMatch(pattern: string, input: string): boolean {
        // Only '*' wildcard, matches any chars within a segment. Exact otherwise.
        if (pattern === '*' || pattern === '**') return true;
        // Build regex: escape regex chars, then replace \* with [^.]*
        const escaped = pattern.replace(/[.+?^${}()|[\]\\]/g, '\\$&').replace(/\*/g, '[^.]*');
        return new RegExp('^' + escaped + '$').test(input);
    }

    // ── WsHub ────────────────────────────────────────────────────────────────
    export class WsHub {
        private readonly audit: AuditChain;
        private readonly gridName: string;
        private readonly bufferCapacity: number;
        private readonly watermarkBytes: number;
        private readonly _clients: Set<ClientConnection> = new Set();
        private readonly unsubscribeAudit: () => void;
        private closing = false;

        constructor(opts: WsHubOptions) {
            this.audit = opts.audit;
            this.gridName = opts.gridName;
            this.bufferCapacity = opts.bufferCapacity ?? DEFAULT_BUFFER_CAPACITY;
            this.watermarkBytes = opts.watermarkBytes ?? DEFAULT_WATERMARK_BYTES;
            // CRITICAL: single subscription. The listener ONLY enqueues.
            this.unsubscribeAudit = this.audit.onAppend((entry) => this.onAuditEvent(entry));
        }

        get clientCount(): number { return this._clients.size; }
        get clients(): ReadonlySet<unknown> { return this._clients; }

        onConnect(socket: ServerSocket, _req?: { headers?: Record<string, unknown> }): void {
            if (this.closing) {
                try { socket.close(1001, 'shutting down'); } catch { /* swallow */ }
                return;
            }
            const client = new ClientConnection(socket, this.watermarkBytes, this.bufferCapacity);
            this._clients.add(client);

            // Send Hello immediately (direct, no buffering).
            const hello: HelloFrame = {
                type: 'hello',
                serverTime: Date.now(),
                gridName: this.gridName,
                lastEntryId: this.audit.length,
            };
            try { socket.send(JSON.stringify(hello)); } catch { /* swallow */ }

            socket.on('message', (raw) => client.handleClientMessage(raw));
            socket.on('close', () => {
                client.markClosed();
                this._clients.delete(client);
            });
            socket.on('error', () => {
                client.markClosed();
                this._clients.delete(client);
            });
        }

        /**
         * Called from the audit listener. Never awaits. Never throws.
         * Enforces the broadcast allowlist at the single fan-out point.
         */
        private onAuditEvent(entry: AuditEntry): void {
            if (!isAllowlisted(entry.eventType)) return;
            for (const client of this._clients) {
                if (!client.matches(entry.eventType)) continue;
                try {
                    client.enqueue(entry);
                } catch {
                    // Swallow — a bug in enqueue must never crash append().
                }
            }
        }

        async close(): Promise<void> {
            if (this.closing) return;
            this.closing = true;
            // Unsubscribe first so no new events arrive during drain.
            try { this.unsubscribeAudit(); } catch { /* swallow */ }

            for (const client of this._clients) {
                client.sendBye('shutting down');
                try { client.socket.close(1001, 'shutting down'); } catch { /* swallow */ }
                client.markClosed();
            }
            this._clients.clear();
        }
    }
    ```
