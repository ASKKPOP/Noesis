    ```typescript
    import { describe, it, expect, beforeAll } from 'vitest';
    import { RelationshipListener } from '../../src/relationships/listener';
    import { AuditChain } from '../../src/audit';
    import { makeTestAudit } from '../helpers';  // existing helper

    describe('REL-04 10K-edge perf bench', () => {
      let listener: RelationshipListener;
      const TAU = 1000;
      const N = 5;
      const NODE_COUNT = 200;     // 200*199/2 = 19900 potential edges; we materialize 10K
      const EDGE_COUNT = 10_000;
      const ITERATIONS = 1_000;

      beforeAll(() => {
        const audit = makeTestAudit();
        listener = new RelationshipListener(audit, { tau: TAU, snapshotCadenceTicks: 100 });

        // Seed 10K edges via synthesized audit events.
        // Use deterministic seeded PRNG (seedrandom or xoshiro in test helpers; D-9-12 bans Math.random in src/** only, tests can use a seeded PRNG).
        const rng = makeSeededRNG(42);
        for (let i = 0; i < EDGE_COUNT; i++) {
          const a = `did:nous:grid:${String(Math.floor(rng() * NODE_COUNT)).padStart(4, '0')}`;
          const b = `did:nous:grid:${String(Math.floor(rng() * NODE_COUNT)).padStart(4, '0')}`;
          if (a === b) { i--; continue; }  // skip self-loops (D-9-11)
          audit.append('nous.spoke', { speaker_did: a, listener_did: b, tick: i }, i);
        }
      });

      it('getTopNFor p95 < 100ms over 1000 iterations at tick 2000', () => {
        const currentTick = 2000;
        const timings: number[] = [];

        // Cycle through ~200 dids so we don't hit the same hot cache line every call.
        for (let i = 0; i < ITERATIONS; i++) {
          const did = `did:nous:grid:${String(i % NODE_COUNT).padStart(4, '0')}`;
          const start = process.hrtime.bigint();
          listener.getTopNFor(did, N, currentTick);
          const end = process.hrtime.bigint();
          timings.push(Number(end - start) / 1_000_000);  // ns -> ms
        }

        timings.sort((a, b) => a - b);
        const p95 = timings[Math.floor(ITERATIONS * 0.95)];
        const p99 = timings[Math.floor(ITERATIONS * 0.99)];

        // Log for visibility even on pass:
        console.log(`perf-10k: p50=${timings[500].toFixed(2)}ms p95=${p95.toFixed(2)}ms p99=${p99.toFixed(2)}ms`);

        expect(p95).toBeLessThan(100);
      });
    });
    ```

    **grid/test/relationships/idempotent-rebuild.test.ts** — REL-02:

    ```typescript
    import { describe, it, expect } from 'vitest';
    import { AuditChain } from '../../src/audit';
    import { RelationshipListener } from '../../src/relationships/listener';
    import { canonicalEdge } from '../../src/relationships/canonical';
    import { makeFixedEventSequence } from '../helpers';

    describe('REL-02 idempotent rebuild', () => {
      it('two rebuilds produce byte-identical canonical snapshots', () => {
        const audit = new AuditChain({ /* test config */ });
        for (const e of makeFixedEventSequence(500)) {
          audit.append(e.type, e.payload, e.tick);
        }

        const listener = new RelationshipListener(audit, { tau: 1000, snapshotCadenceTicks: 100 });
        listener.rebuildFromChain();
        const snapshotA = snapshotCanonical(listener);

        listener.rebuildFromChain();  // second rebuild — MUST NOT drift
        const snapshotB = snapshotCanonical(listener);

        expect(snapshotB).toBe(snapshotA);  // byte-identical JSON string

        // Additional: compare against a fresh listener that runs rebuild only once.
        const fresh = new RelationshipListener(audit, { tau: 1000, snapshotCadenceTicks: 100 });
        fresh.rebuildFromChain();
        expect(snapshotCanonical(fresh)).toBe(snapshotA);
      });

      it('rebuild is idempotent under P-9-03 float drift', () => {
        // Guard: assert canonicalEdge with toFixed(3) absorbs any Math.exp drift
        // by constructing two edges whose weight differs in the 5th decimal
        // and asserting canonicalEdge returns the same string.
        const edge1 = { did_a: 'did:nous:grid:0001', did_b: 'did:nous:grid:0002', valence: 0.123456, weight: 0.567891, recency_tick: 100, last_event_hash: 'a'.repeat(64) };
        const edge2 = { ...edge1, valence: 0.123459, weight: 0.567894 };
        expect(canonicalEdge(edge1)).toBe(canonicalEdge(edge2));
      });
    });

    function snapshotCanonical(listener: RelationshipListener): string {
      const edges = Array.from(listener.edges.values());
      edges.sort((a, b) => {
        const ka = `${a.did_a}|${a.did_b}`;
        const kb = `${b.did_a}|${b.did_b}`;
        return ka.localeCompare(kb);
      });
      return JSON.stringify(edges.map(canonicalEdge));
    }
    ```
