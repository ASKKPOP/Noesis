    ```typescript
    import useSWR from 'swr';
    import { useTick } from '@/lib/stores/tick-store';  // or wherever currentTick lives
    import { fetchRelationshipsH1, fetchRelationshipsH2, fetchGraph } from '@/lib/api/relationships';

    const BATCH_WINDOW_TICKS = 100;  // D-9-13 — NEVER change without phase decision

    export function useRelationshipsH1(did: string | null) {
      const currentTick = useTick();
      const windowKey = Math.floor(currentTick / BATCH_WINDOW_TICKS);
      return useSWR(
        did ? ['relationships-h1', did, windowKey] : null,
        () => fetchRelationshipsH1(did!),
        { revalidateOnFocus: false, dedupingInterval: 0 },
      );
    }

    export function useRelationshipsH2(did: string | null, tier: 'H1' | 'H2' | 'H5') {
      const currentTick = useTick();
      const windowKey = Math.floor(currentTick / BATCH_WINDOW_TICKS);
      return useSWR(
        did && (tier === 'H2' || tier === 'H5') ? ['relationships-h2', did, windowKey] : null,
        () => fetchRelationshipsH2(did!, 'H2'),
        { revalidateOnFocus: false, dedupingInterval: 0 },
      );
    }

    export function useGraph() {
      const currentTick = useTick();
      const windowKey = Math.floor(currentTick / BATCH_WINDOW_TICKS);
      return useSWR(
        ['graph', windowKey],
        () => fetchGraph(),
        { revalidateOnFocus: false, dedupingInterval: 0 },
      );
    }
    ```

    **dashboard/src/app/grid/relationships-graph/relationship-graph.tsx**:

    ```typescript
    'use client';
    import { useGraph } from '@/lib/hooks/use-relationships';

    const WARMTH_COLOR = {
      cold: '#9ca3af',
      warm: '#f59e0b',
      hot:  '#e11d48',
    } as const;

    const VIEWPORT = { width: 1000, height: 1000 };
    const NODE_RADIUS = 3;
    const EDGE_STROKE_WIDTH = 0.5;

    export function RelationshipGraph() {
      const { data, error, isLoading } = useGraph();

      if (isLoading) return <div role="status">Loading graph…</div>;
      if (error) return <div role="alert">Graph could not be loaded.</div>;
      if (!data) return null;

      // Build a lookup: did → {x,y} for edge endpoint resolution.
      const nodePos = new Map(data.nodes.map(n => [n.did, n]));

      return (
        <svg
          viewBox={`0 0 ${VIEWPORT.width} ${VIEWPORT.height}`}
          role="img"
          aria-label="Relationship graph with {data.nodes.length} nodes and {data.edges.length} edges"
        >
          <g class="edges">
            {data.edges.map((e, i) => {
              const a = nodePos.get(e.source_did);
              const b = nodePos.get(e.target_did);
              if (!a || !b) return null;
              return (
                <line
                  key={i}
                  x1={a.x} y1={a.y} x2={b.x} y2={b.y}
                  stroke={WARMTH_COLOR[e.warmth_bucket]}
                  strokeWidth={EDGE_STROKE_WIDTH}
                />
              );
            })}
          </g>
          <g class="nodes">
            {data.nodes.map(n => (
              <circle key={n.did} cx={n.x} cy={n.y} r={NODE_RADIUS} fill="#333" />
            ))}
          </g>
        </svg>
      );
    }
    ```

    The render is O(nodes + edges) with zero layout math on the client. For
    10K edges, React's reconciliation is the only cost. Use a `key` derived
    from a stable identifier (edge source+target, or index). If React render

    `<path d="M x1 y1 L x2 y2 M x3 y3 L x4 y4 ...">` — but attempt keyed
    `<line>` first and only fall back if the Wave 4 perf bench fails.
