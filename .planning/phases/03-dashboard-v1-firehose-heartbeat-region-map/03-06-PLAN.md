    Create `dashboard/src/app/grid/components/region-map.tsx` as a client component (`'use client';`) exporting `RegionMap({ regions, connections }: { regions: Region[]; connections: RegionConnection[] })`.

    Implementation spec:
    ```tsx
    'use client';

    import { memo, useMemo } from 'react';
    import type { Region, RegionConnection } from '@/lib/protocol/region-types';
    import { usePresence } from '../hooks';

    const VIEWPORT_W = 720;
    const VIEWPORT_H = 480;
    const REGION_R = 16;
    const MARKER_R = 5;

    function project(region: Region): { cx: number; cy: number } {
      return { cx: region.x * VIEWPORT_W, cy: region.y * VIEWPORT_H };
    }

    export const RegionMap = memo(function RegionMap({
      regions,
      connections,
    }: {
      regions: Region[];
      connections: RegionConnection[];
    }) {
      const presence = usePresence();

      const regionById = useMemo(() => {
        const m = new Map<string, Region>();
        for (const r of regions) m.set(r.id, r);
        return m;
      }, [regions]);

      const edges = useMemo(() => {
        return connections
          .map((c) => {
            const from = regionById.get(c.from);
            const to = regionById.get(c.to);
            if (!from || !to) return null;
            const a = project(from);
            const b = project(to);
            return { key: `${c.from}→${c.to}`, ...a, bx: b.cx, by: b.cy };
          })
          .filter((e): e is NonNullable<typeof e> => e !== null);
      }, [connections, regionById]);

      return (
        <svg
          viewBox={`0 0 ${VIEWPORT_W} ${VIEWPORT_H}`}
          role="img"
          aria-label="Region map"
          className="w-full h-full bg-neutral-900 border border-neutral-800 rounded-md"
        >
          {regions.length === 0 && (
            <text x={VIEWPORT_W / 2} y={VIEWPORT_H / 2} textAnchor="middle" className="sr-only">
              No regions loaded
            </text>
          )}

          {/* Edges rendered first so they sit under region circles */}
          <g data-layer="edges">
            {edges.map((e) => (
              <line
                key={e.key}
                data-edge={e.key}
                x1={e.cx}
                y1={e.cy}
                x2={e.bx}
                y2={e.by}
                stroke="#404040"
                strokeWidth={1}
              />
            ))}
          </g>

          {/* Regions */}
          <g data-layer="regions">
            {regions.map((r) => {
              const { cx, cy } = project(r);
              return (
                <g key={r.id}>
                  <circle
                    data-region-id={r.id}
                    cx={cx}
                    cy={cy}
                    r={REGION_R}
                    fill="#171717"
                    stroke="#525252"
                    strokeWidth={1.5}
                  />
                  <text
                    x={cx}
                    y={cy + REGION_R + 14}
                    textAnchor="middle"
                    fill="#a3a3a3"
                    fontSize={11}
                    fontFamily="Inter, system-ui, sans-serif"
                  >
                    {r.name}
                  </text>
                </g>
              );
            })}
          </g>

          {/* Nous markers — keyed by DID so React preserves nodes across moves */}
          <g data-layer="markers">
            {Array.from(presence.nousByDid.entries()).map(([did, info]) => {
              const region = regionById.get(info.regionId);
              if (!region) return null;
              const { cx, cy } = project(region);
              return (
                <g
                  key={did}
                  data-did={did}
                  style={{
                    transform: `translate(${cx}px, ${cy}px)`,
                    transition: 'transform 150ms ease-out',
                  }}
                >
                  <circle r={MARKER_R} fill="#10b981" />
                  <title>{info.name}</title>
                </g>
              );
            })}
          </g>
        </svg>
      );
    });
    ```

    Rationale:
    - `memo` prevents re-render when props are referentially equal (presence updates re-render via hook subscription, not prop change).
    - Keying markers by DID ensures React reuses the same DOM node when a Nous moves — CSS `transition: transform` then animates between coordinates (MAP-03).
    - `data-region-id` / `data-edge` / `data-did` attributes give the component tests and Playwright smoke stable selectors.
    - `aria-label="Region map"` + `role="img"` satisfies MAP-01 accessibility; visually-hidden empty-state text gives screen-reader feedback.

    import Fastify from 'fastify';
    import websocketPlugin from '@fastify/websocket';
    import type { WebSocket } from 'ws';

    const REGIONS = [
      { id: 'region-a', name: 'Atrium',  x: 0.2, y: 0.3 },
      { id: 'region-b', name: 'Belfry',  x: 0.7, y: 0.4 },
      { id: 'region-c', name: 'Cloister', x: 0.5, y: 0.8 },
    ];
    const CONNECTIONS = [
      { from: 'region-a', to: 'region-b' },
      { from: 'region-b', to: 'region-c' },
    ];

    export async function startMockGrid(port = 8080) {
      const app = Fastify({ logger: false });
      await app.register(websocketPlugin);

      app.get('/api/v1/grid/regions', async () => ({
        regions: REGIONS,
        connections: CONNECTIONS,
      }));

      const sockets = new Set<WebSocket>();

      app.get('/ws/events', { websocket: true }, (conn) => {
        const socket = conn.socket ?? conn;
        sockets.add(socket);

        const send = (obj: unknown) => socket.send(JSON.stringify(obj));

        send({ type: 'hello', serverTime: Date.now(), latestId: 0 });

        // Send scripted frames on a short delay so the client registers listeners first.
        setTimeout(() => {
          send({
            type: 'event',
            entry: {
              id: 1,
              ts: Date.now(),
              eventType: 'nous.spawned',
              actorDid: 'did:example:alice',
              payload: { name: 'Alice', regionId: 'region-a' },
            },
          });

          for (let t = 1; t <= 3; t++) {
            send({
              type: 'event',
              entry: {
                id: 1 + t,
                ts: Date.now(),
                eventType: 'tick',
                actorDid: 'system',
                payload: { tick: t, epoch: 0, tickRateMs: 1000, timestamp: Date.now() },
              },
            });
          }

          send({
            type: 'event',
            entry: {
              id: 5,
              ts: Date.now(),
              eventType: 'nous.moved',
              actorDid: 'did:example:alice',
              payload: {
                name: 'Alice',
                fromRegion: 'region-a',
                toRegion: 'region-b',
                travelCost: 1,
                tick: 4,
              },
            },
          });
        }, 200);

        socket.on('close', () => sockets.delete(socket));
      });

      await app.listen({ port, host: '127.0.0.1' });
      return {
        url: `http://127.0.0.1:${port}`,
        stop: async () => {
          for (const s of sockets) s.close();
          await app.close();
        },
      };
    }
