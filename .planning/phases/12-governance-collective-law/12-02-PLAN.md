```ts
// grid/src/governance/appendProposalOpened.ts
//
// SOLE PRODUCER for `proposal.opened` audit event. Per D-12-01 / VOTE-01:
//   This is the ONLY file in grid/src/** that may call `audit.append('proposal.opened', ...)`.
//   The grep gate `grid/test/governance/governance-producer-boundary.test.ts` enforces this contract.
//
// Forbidden siblings (any of these in grid/src/** = test failure):
//   proposal.created, proposal.draft, proposal.submitted, governance.opened
//
// Privacy: body text NEVER appears in the audit payload. Only `title_hash = sha256(body)[:32]`.
//   Forbidden keys (rejected with throw): text, body, content, description, rationale,
//   proposal_text, law_text, body_text, weight, reputation, relationship_score, ousia_weight.
//
// Wall-clock ban: no Date.now / Math.random in this file. Tick is provided by caller.
//
// Tier: this emitter does NOT enforce tier gates — that is the API route's responsibility.
//   Operators are forbidden from CALLING this function via grep gate `check-governance-isolation.mjs`.

import { createHash, randomUUID } from 'node:crypto';
import { audit } from '../audit/audit-chain';
import {
  PROPOSAL_OPENED_KEYS,
  GOVERNANCE_FORBIDDEN_KEYS,
  type ProposalOpenedPayload,
} from './types';
import { GOVERNANCE_CONFIG } from './config';
import type { GovernanceStore } from './store';

const DID_RE = /^did:noesis:[a-z0-9_\-]+$/i;

export interface AppendProposalOpenedInput {
  proposer_did: string;
  body_text: string;            // stored in MySQL, NEVER in audit payload
  quorum_pct?: number;          // default 50
  supermajority_pct?: number;   // default 67
  deadline_tick: number;
  currentTick: number;          // for opened_at_tick column
  store: GovernanceStore;
}

export interface AppendProposalOpenedResult {
  proposal_id: string;
  title_hash: string;
}

export async function appendProposalOpened(input: AppendProposalOpenedInput): Promise<AppendProposalOpenedResult> {
  // 1. Validate inputs (DID, tick non-negative, body non-empty)
  if (!DID_RE.test(input.proposer_did)) {
    throw new Error('proposal.opened: proposer_did fails DID_RE');
  }
  if (!Number.isInteger(input.deadline_tick) || input.deadline_tick <= input.currentTick) {
    throw new Error('proposal.opened: deadline_tick must be a future integer tick');
  }
  if (typeof input.body_text !== 'string' || input.body_text.length === 0) {
    throw new Error('proposal.opened: body_text must be non-empty string');
  }

  // 2. Compute title_hash (sha256 truncated to 32 chars)
  const title_hash = createHash('sha256').update(input.body_text).digest('hex').slice(0, 32);

  // 3. Generate proposal_id (UUID v4 from node:crypto)
  const proposal_id = randomUUID();

  // 4. Build closed payload — NOTE: body_text is NOT in this payload
  const quorum_pct = input.quorum_pct ?? GOVERNANCE_CONFIG.quorumPctDefault;
  const supermajority_pct = input.supermajority_pct ?? GOVERNANCE_CONFIG.supermajorityPctDefault;
  if (!Number.isInteger(quorum_pct) || quorum_pct < 1 || quorum_pct > 100) {
    throw new Error('proposal.opened: quorum_pct must be 1..100');
  }
  if (!Number.isInteger(supermajority_pct) || supermajority_pct < 1 || supermajority_pct > 100) {
    throw new Error('proposal.opened: supermajority_pct must be 1..100');
  }
  const payload: ProposalOpenedPayload = {
    deadline_tick: input.deadline_tick,
    proposal_id,
    proposer_did: input.proposer_did,
    quorum_pct,
    supermajority_pct,
    title_hash,
  };

  // 5. Closed-tuple shape check
  const actualKeys = Object.keys(payload).sort();
  const expectedKeys = [...PROPOSAL_OPENED_KEYS].sort();
  if (actualKeys.length !== expectedKeys.length || !actualKeys.every((k, i) => k === expectedKeys[i])) {
    throw new Error(`proposal.opened: payload keys ${JSON.stringify(actualKeys)} != ${JSON.stringify(expectedKeys)}`);
  }

  // 6. Forbidden-key check
  for (const k of actualKeys) {
    if (GOVERNANCE_FORBIDDEN_KEYS.includes(k as typeof GOVERNANCE_FORBIDDEN_KEYS[number])) {
      throw new Error(`proposal.opened: forbidden key "${k}" in payload`);
    }
  }

  // 7. DB write FIRST (so failed audit append leaves no orphan; or use a transaction wrapper if existing pattern)
  await input.store.insertProposal({
    proposal_id,
    proposer_did: input.proposer_did,
    title_hash,
    body_text: input.body_text,
    quorum_pct,
    supermajority_pct,
    deadline_tick: input.deadline_tick,
    opened_at_tick: input.currentTick,
  });

  // 8. Audit append (sole-producer line — grep gate scans for this exact pattern)
  await audit.append('proposal.opened', payload);

  return { proposal_id, title_hash };
}
```

```ts
// grid/src/governance/tally.ts
//
// Pure function — NO I/O, NO audit.append, NO DB. Caller passes hydrated state.
// Per D-12-03: pessimistic quorum (committed-but-unrevealed counts toward denominator).

import type { BallotRevealedPayload, ProposalOutcome } from './types';

export interface TallyInput {
  reveals: ReadonlyArray<Pick<BallotRevealedPayload, 'choice' | 'voter_did'>>;
  committedDids: ReadonlyArray<string>;          // ALL committed DIDs (revealed + unrevealed)
  totalNousCount: number;
  quorumPct: number;          // 1..100
  supermajorityPct: number;   // 1..100
}

export interface TallyResult {
  abstain_count: number;
  no_count: number;
  outcome: ProposalOutcome;
  quorum_met: boolean;
  yes_count: number;
}

export function computeTally(input: TallyInput): TallyResult {
  // Validate
  if (!Number.isInteger(input.totalNousCount) || input.totalNousCount < 0) {
    throw new Error('computeTally: totalNousCount must be non-negative integer');
  }
  if (input.quorumPct < 1 || input.quorumPct > 100) throw new Error('computeTally: quorumPct out of range');
  if (input.supermajorityPct < 1 || input.supermajorityPct > 100) throw new Error('computeTally: supermajorityPct out of range');

  // Count vote types from reveals
  let yes_count = 0;
  let no_count = 0;
  let abstain_count = 0;
  for (const r of input.reveals) {
    if (r.choice === 'yes') yes_count++;
    else if (r.choice === 'no') no_count++;
    else if (r.choice === 'abstain') abstain_count++;
  }

  // Pessimistic quorum: revealed + committed-but-unrevealed
  const committedSet = new Set(input.committedDids);
  // committedDids includes revealed voters too — use set size as the participation count
  const participation = committedSet.size;
  const quorumThreshold = Math.ceil((input.quorumPct / 100) * input.totalNousCount);
  const quorum_met = input.totalNousCount === 0 ? false : participation >= quorumThreshold;

  // Outcome
  let outcome: ProposalOutcome;
  if (!quorum_met) {
    outcome = 'quorum_fail';
  } else {
    const decisive = yes_count + no_count;  // abstain does not count toward supermajority denominator
    if (decisive === 0) {
      outcome = 'rejected';
    } else {
      const yesRatio = yes_count / decisive;
      const supermajorityThreshold = input.supermajorityPct / 100;
      outcome = yesRatio >= supermajorityThreshold ? 'passed' : 'rejected';
    }
  }

  return { abstain_count, no_count, outcome, quorum_met, yes_count };
}
```

```ts
import type { GovernanceStore } from './store';
import type { NousRegistry } from '../registry/nous-registry';
import { appendProposalTallied } from './appendProposalTallied';

export class GovernanceEngine {
  constructor(
    private readonly store: GovernanceStore,
    private readonly registry: NousRegistry,
  ) {}

  /** Called by GenesisLauncher's clock.onTick (W3 wires this). */
  async onTickClosed(currentTick: number): Promise<void> {
    const open = await this.store.getOpenProposals();
    for (const p of open) {
      if (currentTick >= p.deadline_tick) {
        await appendProposalTallied({
          proposal_id: p.proposal_id,
          currentTick,
          store: this.store,
          registry: this.registry,
        });
      }
    }
  }
}
```

```ts
import * as path from 'node:path';

export interface Fixture {
  totalNousCount: number;
  proposer_did: string;
  body_text: string;
  deadline_tick: number;
  startTick: number;
  ballots: Array<{
    voter_did: string;
    nonce: string;
    choice: 'yes' | 'no' | 'abstain';
    revealed: boolean;
  }>;

export async function runFixtureReplay(fixtureId: string): Promise<string> {
  const fixturePath = path.join(__dirname, '..', '..', 'test', 'governance', 'fixtures', `${fixtureId}.json`);
  const fixture: Fixture = JSON.parse(fs.readFileSync(fixturePath, 'utf-8'));

  const store = createInMemoryStore();
  const registry = createInMemoryRegistry({ totalNousCount: fixture.totalNousCount });
  const engine = new GovernanceEngine(store, registry);

  // Reset audit chain to empty for deterministic hashing
  await audit.reset();

  // 1. Open proposal
  const { proposal_id } = await appendProposalOpened({
    proposer_did: fixture.proposer_did,
    body_text: fixture.body_text,
    deadline_tick: fixture.deadline_tick,
    currentTick: fixture.startTick,
    store,
  });

  // 2. Commit each ballot
  let tick = fixture.startTick + 1;
  for (const b of fixture.ballots) {
    const commit_hash = computeCommitHash(b.choice, b.nonce, b.voter_did);
    await appendBallotCommitted({ proposal_id, voter_did: b.voter_did, commit_hash, currentTick: tick, store });
    tick++;
  }

  // 3. Reveal flagged ballots
  for (const b of fixture.ballots) {
    if (b.revealed) {
      await appendBallotRevealed({ proposal_id, voter_did: b.voter_did, choice: b.choice, nonce: b.nonce, currentTick: tick, store });
      tick++;
    }
  }

  // 4. Trigger tally at deadline
  await engine.onTickClosed(fixture.deadline_tick);

  return getChainTailHash();
}
```

```ts
import { audit } from '../audit/audit-chain';

import { LAW_TRIGGERED_KEYS, type LawTriggeredPayload, type Law } from '../logos/types';
import { GOVERNANCE_FORBIDDEN_KEYS } from './types';

export interface AppendLawTriggeredInput {
  law: Law;
  enacted_by: 'collective' | 'operator';
  currentTick: number;
  engine: LogosEngine;

export async function appendLawTriggered(input: AppendLawTriggeredInput): Promise<void> {
  // 1. Add law to LogosEngine
  await input.engine.addLaw(input.law);

  // 2. Build closed payload
  const payload: LawTriggeredPayload = {
    enacted_by: input.enacted_by,
    law_hash: input.law.hash,        // adapt to actual Law shape after reading types.ts
    law_id: input.law.id,            // adapt
    triggered_at_tick: input.currentTick,
  };

  // 3. Closed-tuple shape check
  const actualKeys = Object.keys(payload).sort();
  const expectedKeys = [...LAW_TRIGGERED_KEYS].sort();
  if (actualKeys.length !== expectedKeys.length || !actualKeys.every((k, i) => k === expectedKeys[i])) {
    throw new Error(`law.triggered: payload keys ${JSON.stringify(actualKeys)} != ${JSON.stringify(expectedKeys)}`);
  }

  // 4. Forbidden-key check (vote-weighting keys must never appear here either, per VOTE-06 hygiene)
  for (const k of actualKeys) {
    if (GOVERNANCE_FORBIDDEN_KEYS.includes(k as typeof GOVERNANCE_FORBIDDEN_KEYS[number])) {
      throw new Error(`law.triggered: forbidden key "${k}" in payload`);
    }
  }

  // 5. Audit append (sole producer line)
  await audit.append('law.triggered', payload);
}
```
