```ts
import { createHash } from 'node:crypto';
import type { BallotChoice } from './types';

// SYNC: brain/src/noesis_brain/governance/commit_reveal.py — formula MUST be identical
// Per D-12-02: commit_hash = sha256(choice + '|' + nonce + '|' + voter_did)
// Pipe delimiters prevent chosen-plaintext ambiguity (e.g. choice='yes_x', nonce='abc' vs choice='yes', nonce='_xabc')

const NONCE_HEX_LEN = 32;        // 16 bytes hex
const CHOICE_VALUES = ['yes', 'no', 'abstain'] as const;
const DID_RE = /^did:noesis:[a-z0-9_\-]+$/i;

/** Compute sha256(choice|nonce|voter_did) and return lowercase hex (64 chars). */
export function computeCommitHash(
  choice: BallotChoice,
  nonce: string,
  voter_did: string,
): string {
  if (!CHOICE_VALUES.includes(choice as typeof CHOICE_VALUES[number])) {
    throw new Error(`commit-reveal: invalid choice "${choice}"`);
  }
  if (typeof nonce !== 'string' || nonce.length !== NONCE_HEX_LEN || !/^[0-9a-f]+$/i.test(nonce)) {
    throw new Error(`commit-reveal: nonce must be ${NONCE_HEX_LEN} lowercase hex chars`);
  }
  if (!DID_RE.test(voter_did)) {
    throw new Error(`commit-reveal: voter_did fails DID_RE`);
  }
  const input = `${choice}|${nonce.toLowerCase()}|${voter_did}`;
  return createHash('sha256').update(input).digest('hex');
}

/** Verify a revealed ballot against the originally committed hash. Returns boolean; never throws on mismatch. */
export function verifyReveal(
  choice: BallotChoice,
  nonce: string,
  voter_did: string,
  expected_commit_hash: string,
): boolean {
  try {
    const actual = computeCommitHash(choice, nonce, voter_did);
    return constantTimeEqualHex(actual, expected_commit_hash);
  } catch {
    return false;
  }
}

/** Constant-time hex string comparison (defense against timing oracles, even though commit_hash is non-secret). */
function constantTimeEqualHex(a: string, b: string): boolean {
  if (a.length !== b.length) return false;
  let diff = 0;
  for (let i = 0; i < a.length; i++) diff |= a.charCodeAt(i) ^ b.charCodeAt(i);
  return diff === 0;
}
```

```python
"""Commit-reveal hashing for governance ballots.

SYNC: grid/src/governance/commit-reveal.ts — formula MUST be identical
Per D-12-02: commit_hash = sha256(choice + '|' + nonce + '|' + voter_did)
"""
import hashlib
import hmac
import re
import secrets
from typing import Literal

BallotChoice = Literal["yes", "no", "abstain"]
NONCE_HEX_LEN = 32  # 16 bytes hex
DID_RE = re.compile(r"^did:noesis:[a-z0-9_\-]+$", re.IGNORECASE)
CHOICE_VALUES = ("yes", "no", "abstain")


def generate_nonce() -> str:
    """Generate 32 lowercase hex chars (16 bytes) using secrets.token_hex.

    secrets.token_hex is the documented cryptographically-secure source per
    Python stdlib — it wraps os.urandom internally. Do NOT substitute random.random
    or random.randint (predictable PRNG).
    """
    return secrets.token_hex(NONCE_HEX_LEN // 2)


def compute_commit_hash(choice: BallotChoice, nonce: str, voter_did: str) -> str:
    """Return sha256(choice|nonce|voter_did) as lowercase hex (64 chars)."""
    if choice not in CHOICE_VALUES:
        raise ValueError(f"commit-reveal: invalid choice {choice!r}")
    if not isinstance(nonce, str) or len(nonce) != NONCE_HEX_LEN or not re.fullmatch(r"[0-9a-fA-F]+", nonce):
        raise ValueError(f"commit-reveal: nonce must be {NONCE_HEX_LEN} hex chars")
    if not DID_RE.match(voter_did):
        raise ValueError("commit-reveal: voter_did fails DID_RE")
    payload = f"{choice}|{nonce.lower()}|{voter_did}"
    return hashlib.sha256(payload.encode("utf-8")).hexdigest()


def verify_reveal(choice: BallotChoice, nonce: str, voter_did: str, expected_commit_hash: str) -> bool:
    """Return True iff sha256(choice|nonce|voter_did) == expected_commit_hash. Constant-time. Never raises on mismatch."""
    try:
        actual = compute_commit_hash(choice, nonce, voter_did)
    except ValueError:
        return False
    return hmac.compare_digest(actual, expected_commit_hash)
```

```python
    fixed_upper = fixed_lower.upper()
    assert compute_commit_hash("yes", fixed_lower, "did:noesis:alice") == compute_commit_hash("yes", fixed_upper, "did:noesis:alice")
```
