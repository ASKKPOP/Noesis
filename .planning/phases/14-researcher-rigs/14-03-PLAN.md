    ```javascript
    /**
     * Rig TOML config loader — Phase 14 RIG-01.
     * Parses a TOML config file into a fully-typed RigConfig. Enforces D-14-03 (mutually
     * exclusive nous_manifest vs nous_manifest_path). Resolves external manifest path
     * relative to the TOML file's directory.
     */
    import { parse as parseToml } from 'smol-toml';
    import { readFileSync, existsSync } from 'node:fs';
    import { dirname, isAbsolute, resolve } from 'node:path';

    const REQUIRED_FIELDS = ['seed', 'config_name', 'tick_budget', 'operator_tier_cap'];
    const VALID_TIER_CAPS = ['H1', 'H2', 'H3', 'H4', 'H5'];

    function snakeToCamel(s) {
        return s.replace(/_([a-z])/g, (_, c) => c.toUpperCase());
    }

    function normalizeManifestEntry(raw, lineNo, source) {
        const required = ['name', 'did', 'public_key', 'region'];
        for (const f of required) {
            if (raw[f] === undefined) {
                throw new Error(`[rig-config-loader] manifest entry ${source}#${lineNo}: missing required field "${f}"`);
            }
        }
        return {
            name: raw.name,
            did: raw.did,
            publicKey: raw.public_key,
            region: raw.region,
            ...(raw.human_owner !== undefined ? { humanOwner: raw.human_owner } : {}),
            ...(raw.personality !== undefined ? { personality: raw.personality } : {}),
        };
    }

    function loadExternalManifest(absPath) {
        if (!existsSync(absPath)) {
            throw new Error(`[rig-config-loader] nous_manifest_path does not exist: ${absPath}`);
        }
        const raw = readFileSync(absPath, 'utf8');
        const entries = [];
        const lines = raw.split('\n');
        for (let i = 0; i < lines.length; i++) {
            const line = lines[i].trim();
            if (!line) continue;
            const parsed = JSON.parse(line);
            entries.push(normalizeManifestEntry(parsed, i + 1, absPath));
        }
        if (entries.length === 0) {
            throw new Error(`[rig-config-loader] nous_manifest_path is empty: ${absPath}`);
        }
        return entries;
    }

    /** @returns {Promise<import('../grid/src/rig/types.ts').RigConfig>} */
    export async function loadRigConfigFromToml(tomlPath) {
        const raw = readFileSync(tomlPath, 'utf8');
        const parsed = parseToml(raw);

        for (const f of REQUIRED_FIELDS) {
            if (parsed[f] === undefined) {
                throw new Error(`[rig-config-loader] missing required field "${f}" in ${tomlPath}`);
            }
        }
        if (!VALID_TIER_CAPS.includes(parsed.operator_tier_cap)) {
            throw new Error(`[rig-config-loader] operator_tier_cap must be one of ${VALID_TIER_CAPS.join(',')}`);
        }
        if (typeof parsed.seed !== 'string' || parsed.seed.length < 8) {
            throw new Error('[rig-config-loader] seed must be a string with length >= 8');
        }
        if (!/^[a-z0-9-]+$/.test(parsed.config_name)) {
            throw new Error(`[rig-config-loader] config_name must match /^[a-z0-9-]+$/, got "${parsed.config_name}"`);
        }
        if (typeof parsed.tick_budget !== 'number' || parsed.tick_budget <= 0) {
            throw new Error('[rig-config-loader] tick_budget must be a positive integer');
        }

        // D-14-03: mutually-exclusive manifest source
        const inline = Array.isArray(parsed.nous_manifest) && parsed.nous_manifest.length > 0;
        const external = typeof parsed.nous_manifest_path === 'string' && parsed.nous_manifest_path.length > 0;
        if (inline && external) {
            throw new Error('[rig-config-loader] nous_manifest and nous_manifest_path are mutually exclusive (D-14-03)');
        }
        if (!inline && !external) {
            throw new Error('[rig-config-loader] one of nous_manifest or nous_manifest_path is required');
        }

        let manifest;
        if (inline) {
            manifest = parsed.nous_manifest.map((e, i) => normalizeManifestEntry(e, i + 1, tomlPath));
        } else {
            const baseDir = dirname(resolve(tomlPath));
            const manifestPath = isAbsolute(parsed.nous_manifest_path)
                ? parsed.nous_manifest_path
                : resolve(baseDir, parsed.nous_manifest_path);
            manifest = loadExternalManifest(manifestPath);
        }

        return {
            seed: parsed.seed,
            configName: parsed.config_name,
            tickBudget: parsed.tick_budget,
            tickRateMs: parsed.tick_rate_ms ?? 0,           // D-14-07
            operatorTierCap: parsed.operator_tier_cap,
            llmFixturePath: parsed.llm_fixture_path,        // optional
            permissive: parsed.permissive ?? false,
            ...(inline
                ? { nousManifest: manifest }
                : { nousManifestPath: parsed.nous_manifest_path, nousManifest: manifest }),
        };
    }
    ```
