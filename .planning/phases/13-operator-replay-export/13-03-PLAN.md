CRITICAL determinism rules (from 13-RESEARCH.md §Pattern 7, §Pitfall 3):
1. Sorted entry order: `entries.sort((a, b) => a.path.localeCompare(b.path))` — paths inside the tarball.
2. Fixed mtime: `0` (Unix epoch).
3. Fixed mode: `0o644` for files (no executables in this format).
4. Fixed uid/gid: `0`/`0` via `portable: true`.
5. Canonical JSON: recursive sorted-key walker; no whitespace; LF line endings; no trailing newline at file end.
6. No PAX headers: `noPax: true`.
7. Raw tar (no gzip) — keeps the determinism contract minimal.

    ```typescript
    /**
     * canonicalStringify — deterministic JSON serializer.
     *
     * Sorts object keys recursively at every nesting depth. Arrays preserve
     * their input order (arrays are intrinsically ordered). No whitespace.
     * Throws on undefined, function, symbol, BigInt, or cyclic references.
     *
     * REPLAY-01 byte-determinism: this IS the canonical-JSON contract for
     * tarball-embedded JSON files. Bumping the algorithm requires bumping
     * `ExportManifest.canonical_json_version`.
     *
     * See: 13-RESEARCH.md §Pattern 7, §Don't Hand-Roll.
     */
    export function canonicalStringify(value: unknown): string {
        const seen = new WeakSet<object>();
        return walk(value, seen);
    }

    function walk(value: unknown, seen: WeakSet<object>): string {
        if (value === null) return 'null';
        const t = typeof value;
        if (t === 'string' || t === 'number' || t === 'boolean') {
            // Numbers: rely on JSON.stringify for IEEE-754 canonical formatting.
            // Strings: rely on JSON.stringify for proper escaping.
            return JSON.stringify(value);
        }
        if (t === 'undefined' || t === 'function' || t === 'symbol' || t === 'bigint') {
            throw new TypeError(`canonicalStringify: unsupported type ${t}`);
        }
        if (Array.isArray(value)) {
            if (seen.has(value)) throw new TypeError('canonicalStringify: cycle detected (array)');
            seen.add(value);
            const parts = value.map((v) => walk(v, seen));
            seen.delete(value);
            return `[${parts.join(',')}]`;
        }
        // Plain object branch.
        if (typeof value !== 'object') throw new TypeError(`canonicalStringify: unexpected type ${t}`);
        if (seen.has(value as object)) throw new TypeError('canonicalStringify: cycle detected (object)');
        seen.add(value as object);
        const keys = Object.keys(value as Record<string, unknown>).sort();
        const parts = keys.map((k) => `${JSON.stringify(k)}:${walk((value as Record<string, unknown>)[k], seen)}`);
        seen.delete(value as object);
        return `{${parts.join(',')}}`;
    }
    ```

    Skeleton:
    ```javascript
    #!/usr/bin/env node
    /**
     * scripts/replay-verify.mjs
     *
     * REPLAY-01 verification CLI.
     *
     * Given a tarball produced by `grid/src/export/tarball-builder.ts`, this
     * tool reads the file, recomputes its SHA-256 hash, parses the embedded
     * `manifest.json` as canonical JSON, and asserts:
     *   sha256(<tar bytes>) === manifest.tarball_hash
     *
     * The two-pass build scheme in tarball-builder.ts ensures the manifest
     * inside the tarball is self-referential: the hash field matches the
     * outer tar bytes containing it. This CLI is the verification side.
     *
     * Exit codes:
     *   0  — verified; hashes match.
     *   1  — verification failed; hashes differ (possible tampering).
     *   2  — tarball is missing one of the four expected entries.
     *   3  — manifest.json could not be parsed as canonical JSON.
     *   4  — manifest.tarball_hash does not match HEX64_RE.
     *   64 — usage error (no path provided).
     *
     * See: 13-RESEARCH.md §Pattern 7, 13-VALIDATION.md (manual verification entry),
     *      grid/src/export/tarball-builder.ts (the producer side).
     */

    import { createHash } from 'node:crypto';
    import { readFileSync, statSync } from 'node:fs';
    import { resolve } from 'node:path';
    import { Parse as TarParse } from 'tar';
    import { Readable } from 'node:stream';

    const HEX64_RE = /^[0-9a-f]{64}$/;
    const REQUIRED_ENTRIES = new Set(['manifest.json', 'slice.jsonl', 'snapshot.end.json', 'snapshot.start.json']);

    async function main() {
        const args = process.argv.slice(2);
        if (args.length !== 1) {
            console.error('Usage: node scripts/replay-verify.mjs <path-to-tarball>');
            process.exit(64);
        }
        const path = resolve(args[0]);
        let stat;
        try { stat = statSync(path); } catch (err) {
            console.error(`replay-verify: cannot stat ${path}: ${err.message}`);
            process.exit(64);
        }
        if (!stat.isFile()) {
            console.error(`replay-verify: not a file: ${path}`);
            process.exit(64);
        }

        const bytes = readFileSync(path);
        const computedHash = createHash('sha256').update(bytes).digest('hex');

        // Parse tar entries in-memory using tar.Parse stream.
        const found = new Map();   // path → Buffer
        await new Promise((resolveP, rejectP) => {
            const stream = Readable.from(bytes).pipe(new TarParse());
            stream.on('entry', (entry) => {
                const chunks = [];
                entry.on('data', (c) => chunks.push(c));
                entry.on('end', () => {
                    if (REQUIRED_ENTRIES.has(entry.path)) {
                        found.set(entry.path, Buffer.concat(chunks));
                    }
                });
                entry.on('error', rejectP);
            });
            stream.on('end', resolveP);
            stream.on('error', rejectP);
        });

        for (const required of REQUIRED_ENTRIES) {
            if (!found.has(required)) {
                console.error(`replay-verify: tarball missing required entry: ${required}`);
                process.exit(2);
            }
        }

        let manifest;
        try {
            manifest = JSON.parse(found.get('manifest.json').toString('utf8'));
        } catch (err) {
            console.error(`replay-verify: manifest.json parse failed: ${err.message}`);
            process.exit(3);
        }
        if (typeof manifest.tarball_hash !== 'string' || !HEX64_RE.test(manifest.tarball_hash)) {
            console.error(`replay-verify: manifest.tarball_hash invalid (not HEX64): ${manifest.tarball_hash}`);
            process.exit(4);
        }

        if (computedHash !== manifest.tarball_hash) {
            console.error(`replay-verify: HASH MISMATCH`);
            console.error(`  computed:  ${computedHash}`);
            console.error(`  manifest:  ${manifest.tarball_hash}`);
            console.error(`  difference suggests tampering or a non-canonical re-pack.`);
            process.exit(1);
        }

        console.log(`replay-verify: ${path} ✓`);
        console.log(`  hash: ${computedHash}`);
        console.log(`  ticks: ${manifest.start_tick}–${manifest.end_tick}, entries: ${manifest.entry_count}`);
        console.log(`  chain_tail: ${manifest.chain_tail_hash}`);
        process.exit(0);
    }

    main().catch((err) => {
        console.error(`replay-verify: unexpected failure: ${err && err.stack ? err.stack : err}`);
        process.exit(1);
    });
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Export producer ↔ Tarball file | The tarball leaves the system once written. From that point any mutation by an attacker must be detectable. The two-pass self-referencing manifest is the integrity proof. |
| Tarball file ↔ replay-verify consumer | The verifier recomputes the hash from the bytes alone. A successful verification proves byte-identity to the original. A failure proves divergence (tampering, corruption, non-canonical re-pack). |
| canonical-JSON serializer ↔ embedded JSON files | The serializer is the substrate. If the serializer has any source of nondeterminism (key insertion order, locale-dependent toString, etc.), every dependent hash drifts. The serializer is the smallest hand-roll surface and the most heavily tested. |
