    /**

    import { AuditChain, AuditEntry } from './chain';
    import { payloadPrivacyCheck } from './broadcast-allowlist';

    /** 64-hex SHA-256 — matches grid/src/audit/state-hash.ts HEX64_RE. */
    export const HEX64_RE = /^[0-9a-f]{64}$/;

    /** DID regex — shared pattern from _validation.ts / append-telos-refined.ts. */
    export const DID_RE = /^did:noesis:[a-z0-9_\-]+$/i;

    /** Closed 5-key payload for operator.nous_deleted (D-25). */
    export interface NousDeletedPayload {
        readonly tier: 'H5';
        readonly action: 'delete';
        readonly operator_id: string;
        readonly target_did: string;
        readonly pre_deletion_state_hash: string;
    }

    /** The 5 keys a nous_deleted payload must carry — nothing more, nothing less. */
    const EXPECTED_KEYS = [
        'action', 'operator_id', 'pre_deletion_state_hash', 'target_did', 'tier',
    ] as const;

    /**
     * Sole producer path for operator.nous_deleted audit events.
     *
     * @throws TypeError if literal guards fail, regex guards fail, the payload
     *   carries an unexpected key, or payloadPrivacyCheck rejects the payload.
     */
    export function appendNousDeleted(
        audit: AuditChain,
        operatorId: string,
        payload: NousDeletedPayload,
    ): AuditEntry {
        // 1. DID guard on operatorId parameter.
        if (typeof operatorId !== 'string' || !DID_RE.test(operatorId)) {
            throw new TypeError(`appendNousDeleted: invalid operatorId (DID_RE failed)`);
        }
        // 2. Type guard on payload.
        if (payload === null || typeof payload !== 'object' || Array.isArray(payload)) {
            throw new TypeError(`appendNousDeleted: payload must be a plain object`);
        }
        // 3. Literal guards — D-25 tier + action are constants, not generic strings.
        if ((payload as { tier?: unknown }).tier !== 'H5') {
            throw new TypeError(`appendNousDeleted: tier must be literal 'H5', got ${JSON.stringify((payload as { tier?: unknown }).tier)}`);
        }
        if ((payload as { action?: unknown }).action !== 'delete') {
            throw new TypeError(`appendNousDeleted: action must be literal 'delete', got ${JSON.stringify((payload as { action?: unknown }).action)}`);
        }
        // 4. Regex guards on the remaining fields.
        if (typeof payload.operator_id !== 'string' || !DID_RE.test(payload.operator_id)) {
            throw new TypeError(`appendNousDeleted: operator_id must match DID_RE`);
        }
        if (payload.operator_id !== operatorId) {
            throw new TypeError(`appendNousDeleted: payload.operator_id must equal operatorId (self-report invariant)`);
        }
        if (typeof payload.target_did !== 'string' || !DID_RE.test(payload.target_did)) {
            throw new TypeError(`appendNousDeleted: target_did must match DID_RE`);
        }
        if (typeof payload.pre_deletion_state_hash !== 'string' || !HEX64_RE.test(payload.pre_deletion_state_hash)) {
            throw new TypeError(`appendNousDeleted: pre_deletion_state_hash must match HEX64_RE`);
        }
        // 5. Closed-tuple structural check.
        const actualKeys = Object.keys(payload).sort();
        if (actualKeys.length !== EXPECTED_KEYS.length
            || !actualKeys.every((k, i) => k === EXPECTED_KEYS[i])) {
            throw new TypeError(
                `appendNousDeleted: unexpected key set — expected ${JSON.stringify(EXPECTED_KEYS)}, got ${JSON.stringify(actualKeys)}`,
            );
        }
        // 6. Explicit reconstruction — no prototype pollution / prototype-inherited keys.
        const cleanPayload = {
            tier: 'H5' as const,
            action: 'delete' as const,
            operator_id: payload.operator_id,
            target_did: payload.target_did,
            pre_deletion_state_hash: payload.pre_deletion_state_hash,
        };
        // 7. Privacy gate — belt-and-suspenders (D-27).
        const privacy = payloadPrivacyCheck(cleanPayload);
        if (!privacy.ok) {
            throw new TypeError(
                `appendNousDeleted: privacy violation — path=${privacy.offendingPath}, keyword=${privacy.offendingKeyword}`,
            );
        }
        // 8. Commit to chain.
        return audit.append('operator.nous_deleted', operatorId, cleanPayload, payload.target_did);
    }

     * hash-state-client — Grid-side client for Brain's POST /nous/{did}/hash_state.
     *
     * Phase 5/6/7 brain-bridge clients established the pattern: fetch with
     * timeout, error-class mapping, 64-hex validation on response body. Phase

     *
     * See: 08-CONTEXT D-08, D-29, 08-PATTERNS §hash-state-client.

    import { HEX64_RE } from '../audit/state-hash';

    export class BrainUnreachableError extends Error {
        readonly statusHint = 503 as const;
    }
    export class BrainUnknownDidError extends Error {
        readonly statusHint = 404 as const;
    }
    export class BrainMalformedResponseError extends Error {
        readonly statusHint = 503 as const;
    }

    export interface FetchStateHashOptions {
        readonly brainBaseUrl: string;
        readonly timeoutMs?: number;
        readonly fetchImpl?: typeof fetch;
    }

    const REQUIRED_KEYS = ['psyche_hash', 'thymos_hash', 'telos_hash', 'memory_stream_hash'] as const;

    export async function fetchStateHashComponents(
        did: string,
        options: FetchStateHashOptions,
    ): Promise<StateHashComponents> {
        const controller = new AbortController();
        const timeout = setTimeout(() => controller.abort(), options.timeoutMs ?? 3000);
        const f = options.fetchImpl ?? fetch;
        let response: Response;
        try {
            response = await f(`${options.brainBaseUrl}/nous/${encodeURIComponent(did)}/hash_state`, {
                method: 'POST',
                signal: controller.signal,
            });
        } catch (err: unknown) {
            const e = err as { name?: string };
            if (e?.name === 'AbortError') {
                throw new BrainUnreachableError(`Brain /hash_state timeout after ${options.timeoutMs ?? 3000}ms`);
            }
            throw new BrainUnreachableError(`Brain /hash_state network error: ${String(err)}`);
        } finally {
            clearTimeout(timeout);
        }

        if (response.status === 404) {
            throw new BrainUnknownDidError(`Brain does not know DID ${did}`);
        }
        if (!response.ok) {
            throw new BrainUnreachableError(`Brain /hash_state returned HTTP ${response.status}`);
        }

        let body: unknown;
        try {
            body = await response.json();
        } catch {
            throw new BrainMalformedResponseError(`Brain /hash_state returned invalid JSON`);
        }

        if (body === null || typeof body !== 'object' || Array.isArray(body)) {
            throw new BrainMalformedResponseError(`Brain /hash_state body must be a plain object`);
        }
        for (const key of REQUIRED_KEYS) {
            const v = (body as Record<string, unknown>)[key];
            if (typeof v !== 'string' || !HEX64_RE.test(v)) {
                throw new BrainMalformedResponseError(`Brain /hash_state body missing or malformed ${key}`);
            }
        }
        const extra = Object.keys(body).filter(k => !REQUIRED_KEYS.includes(k as typeof REQUIRED_KEYS[number]));
        if (extra.length > 0) {
            // D-03 invariant — Brain must NEVER return a 5th composed hash.
            throw new BrainMalformedResponseError(`Brain /hash_state body has unexpected keys: ${extra.join(',')}`);
        }

        const b = body as Record<string, string>;
        return {
            psyche_hash:        b.psyche_hash,
            thymos_hash:        b.thymos_hash,
            telos_hash:         b.telos_hash,
            memory_stream_hash: b.memory_stream_hash,
        };
    }

    /**
     * POST /api/operator/nous/:did/delete — AGENCY-05 H5 Sovereign Operations
     * delete route. Orchestrates: DID_RE guard → tombstoneCheck (410) →
     * registry.get (404) → Brain /hash_state (503) → combineStateHash →
     * registry.tombstone → appendNousDeleted → 200.
     *
     * Order matters (D-30): tombstone BEFORE audit emit. If tombstone throws
     * unexpectedly, no audit event is written — the Nous remains active.
     *
     * See: 08-CONTEXT D-28, D-29, D-30, D-31, D-33.
     */
    import type { Request, Response } from 'express';  // or whatever the project uses
    import { DID_RE } from './_validation';
    import { tombstoneCheck, TombstonedDidError } from '../../registry/tombstone-check';
    import { combineStateHash } from '../../audit/state-hash';
    import { appendNousDeleted } from '../../audit/append-nous-deleted';
    import {
        fetchStateHashComponents,
        BrainUnreachableError,
        BrainUnknownDidError,
        BrainMalformedResponseError,
    } from '../../brain-bridge/hash-state-client';
    import type { NousRegistry } from '../../registry/registry';
    import type { AuditChain } from '../../audit/chain';

    export interface DeleteNousDeps {
        readonly registry: NousRegistry;
        readonly audit: AuditChain;
        readonly coordinator: { despawnNous: (did: string) => void };  // D-14/D-15 runtime teardown
        readonly brainBaseUrl: string;
        readonly currentTick: () => number;
        readonly operatorId: (req: Request) => string | null;  // auth hook
    }

    export function deleteNousHandler(deps: DeleteNousDeps) {
        return async function handler(req: Request, res: Response): Promise<void> {
            // STEP 1: auth — resolve operator_id.
            const operatorId = deps.operatorId(req);
            if (!operatorId) {
                res.status(401).json({ ok: false, error: 'unauthorized', message: 'no operator context' });
                return;
            }

            // STEP 2: DID_RE guard (400).
            const did = String(req.params.did ?? '');
            if (!DID_RE.test(did)) {
                res.status(400).json({ ok: false, error: 'invalid_did', message: `did must match DID_RE, got ${did}` });
                return;
            }

            // STEP 3: tombstoneCheck (410) — idempotent re-delete attempt.
            try {
                tombstoneCheck(deps.registry, did);
            } catch (err: unknown) {
                if (err instanceof TombstonedDidError) {
                    res.status(410).json({
                        ok: false, error: 'gone',
                        message: `DID ${did} was deleted at tick ${err.deletedAtTick}`,
                        deleted_at_tick: err.deletedAtTick,
                    });
                    return;
                }
                throw err;
            }

            // STEP 4: registry existence (404).
            const record = deps.registry.get(did);
            if (!record) {
                res.status(404).json({ ok: false, error: 'unknown_did', message: `DID ${did} not in registry` });
                return;
            }

            // STEP 5: Brain /hash_state RPC.
            let components;
            try {
                components = await fetchStateHashComponents(did, { brainBaseUrl: deps.brainBaseUrl });
            } catch (err: unknown) {
                if (err instanceof BrainUnknownDidError) {
                    res.status(404).json({
                        ok: false, error: 'brain_unknown_did',
                        message: `Brain does not know DID ${did} (registry/Brain desync — refuse to delete)`,
                    });
                    return;
                }
                if (err instanceof BrainMalformedResponseError) {
                    res.status(503).json({
                        ok: false, error: 'brain_malformed_response',
                        message: String((err as Error).message),
                    });
                    return;
                }
                if (err instanceof BrainUnreachableError) {
                    res.status(503).json({
                        ok: false, error: 'brain_unreachable',
                        message: String((err as Error).message),
                    });
                    return;
                }
                throw err;
            }

            // STEP 6: combineStateHash (Grid-side sole-authority composition per D-03).
            let preDeletionStateHash: string;
            try {
                preDeletionStateHash = combineStateHash(components);
            } catch (err: unknown) {
                // Very rare — Brain returned 64-hex values that still fail combine's
                // closed-tuple check (would indicate a grave bug). Surface as 503.
                res.status(503).json({
                    ok: false, error: 'state_hash_composition_failed',
                    message: String((err as Error).message),
                });
                return;
            }

            // STEP 7: Tombstone (D-30: BEFORE audit emit).
            const tick = deps.currentTick();
            try {
                deps.registry.tombstone(did, tick);
            } catch (err: unknown) {
                // Race — registry mutated between STEP 3 and here (another request
                // tombstoned in parallel). Translate to 410.
                res.status(410).json({
                    ok: false, error: 'gone_race',
                    message: `DID ${did} was tombstoned concurrently: ${String((err as Error).message)}`,
                });
                return;
            }

            // STEP 7.5: Runtime teardown (D-14/D-15 / ROADMAP SC#3). Best-effort —
            // tombstone is the source of truth; the NousRunner tick-skip guard is
            // the correctness barrier. Failures are logged but do not abort the
            // audit emit (the Nous is already status='deleted').
            try {
                deps.coordinator.despawnNous(did);
            } catch (err: unknown) {
                // Log loudly; continue to audit emit. Tick-skip guard will block
                // any late tick dispatch that despawn failed to unsubscribe.
                console.error(`[delete-nous] despawnNous failed for ${did}:`, err);
            }

            // STEP 8: Audit emit — sole producer path.
            try {
                deps.audit.append  // no-op reference just to silence unused-import if audit is unused here
                appendNousDeleted(deps.audit, operatorId, {
