    test('SC#4: tier committed at confirm time, not at HTTP arrival', async ({ page }: { page: Page }) => {

        // Capture the POST body so we can assert tier=H4 even if a downgrade fires mid-flight.
        let capturedBody: { tier?: string } | null = null;
        await page.route('**/api/v1/operator/nous/**/telos', async (route) => {
