```typescript
import type { FastifyInstance } from 'fastify';
import { DID_REGEX } from '../server.js';
import type { GridServices, ApiError } from '../types.js';
import { appendOperatorEvent } from '../../audit/operator-events.js';
import { validateTierBody } from './_validation.js';

interface MemoryQueryBody {
    tier: 'H2';
    operator_id: string;
    query?: string;
    limit?: number;
}

export function registerMemoryQueryRoute(app: FastifyInstance, services: GridServices): void {
    app.post<{ Params: { did: string } }>('/api/v1/operator/nous/:did/memory', async (req, reply) => {
        const did = req.params.did;
        if (!DID_REGEX.test(did)) {
            reply.code(400);
            return { error: 'invalid_did' } satisfies ApiError;
        }
        const v = validateTierBody<MemoryQueryBody>(req.body, 'H2');
        if (!v.ok) { reply.code(400); return v.error; }
        const body = v.body;

        const runner = services.getRunner ? services.getRunner(did) : undefined;
        if (!runner) {
            reply.code(404);
            return { error: 'unknown_nous' } satisfies ApiError;
        }
        if (!runner.connected) {
            // D-16: no action took place — do NOT emit operator.inspected.
            reply.code(503);
            return { error: 'brain_unavailable' } satisfies ApiError;
        }
        try {
            const result = await runner.queryMemory({
                query: typeof body.query === 'string' ? body.query : '',
                limit: typeof body.limit === 'number' ? body.limit : 5,
            });
            // Emit AFTER success per D-16.
            appendOperatorEvent(services.audit, 'operator.inspected', body.operator_id, {
                tier: 'H2',
                action: 'inspect',
                operator_id: body.operator_id,
                target_did: did,
            }, did);
            return result;
        } catch (err) {
            req.log.warn({ err, did }, 'operator memory-query: brain call failed');
            reply.code(503);
            return { error: 'brain_unavailable' } satisfies ApiError;
        }
    });
}
```

**5. `grid/src/api/operator/telos-force.ts` (NEW)** — full handler:

```typescript
import type { FastifyInstance } from 'fastify';
import { DID_REGEX } from '../server.js';
import type { GridServices, ApiError } from '../types.js';
import { appendOperatorEvent } from '../../audit/operator-events.js';
import { validateTierBody } from './_validation.js';

interface TelosForceBody {
    tier: 'H4';
    operator_id: string;
    /** New Telos content — shape is {active_goals: [{description, goal_type, ...}, ...]}.
     *  This value leaves the Grid only as far as the bridge layer; the Grid does NOT audit it. */
    new_telos: Record<string, unknown>;
}

const HEX64 = /^[a-f0-9]{64}$/;

export function registerTelosForceRoute(app: FastifyInstance, services: GridServices): void {
    app.post<{ Params: { did: string } }>('/api/v1/operator/nous/:did/telos', async (req, reply) => {
        const did = req.params.did;
        if (!DID_REGEX.test(did)) {
            reply.code(400);
            return { error: 'invalid_did' } satisfies ApiError;
        }
        const v = validateTierBody<TelosForceBody>(req.body, 'H4');
        if (!v.ok) { reply.code(400); return v.error; }
        const body = v.body;
        if (typeof body.new_telos !== 'object' || body.new_telos === null) {
            reply.code(400);
            return { error: 'invalid_new_telos' } satisfies ApiError;
        }

        const runner = services.getRunner ? services.getRunner(did) : undefined;
        if (!runner) {
            reply.code(404);
            return { error: 'unknown_nous' } satisfies ApiError;
        }
        if (!runner.connected) {
            reply.code(503);
            return { error: 'brain_unavailable' } satisfies ApiError;
        }
        try {
            const diff = await runner.forceTelos(body.new_telos);
            // Defensive: Brain MUST return 64-hex hashes. If it doesn't, the payload
            // privacy gate would still reject non-conforming values but we fail loud.
            if (!HEX64.test(diff.telos_hash_before) || !HEX64.test(diff.telos_hash_after)) {
                req.log.warn({ did }, 'operator telos-force: brain returned non-hex telos hashes');
                reply.code(503);
                return { error: 'brain_unavailable' } satisfies ApiError;
            }
            // D-15 / D-19: ONLY hashes leave in the audit payload. No new_telos, no goal contents.
            appendOperatorEvent(services.audit, 'operator.telos_forced', body.operator_id, {
                tier: 'H4',
                action: 'force_telos',
                operator_id: body.operator_id,
                target_did: did,
                telos_hash_before: diff.telos_hash_before,
                telos_hash_after: diff.telos_hash_after,
            }, did);
            // Response mirrors audit — no goal contents.
            return diff;
        } catch (err) {
            req.log.warn({ err, did }, 'operator telos-force: brain call failed');
            reply.code(503);
            return { error: 'brain_unavailable' } satisfies ApiError;
        }
    });
}
```

**7. `brain/src/noesis_brain/rpc/handler.py` — add after `get_state`**:

```python
async def query_memory(self, params: dict[str, Any]) -> dict[str, Any]:
    """H2 Reviewer memory query. Grid-side audit is the source of truth; handler returns normalized entries only.

    params:
        query: search string (empty → recent)
        limit: max entries (default 5, clamp 1..50)
    """
    query = str(params.get("query", ""))
    try:
        limit = int(params.get("limit", 5))
    except (TypeError, ValueError):
        limit = 5
    limit = max(1, min(50, limit))
    if self.memory is None:
        return {"entries": []}
    try:
        if query:
            # memory.about is DID-indexed; memory.recent is tick-indexed. For a free-text query,
            # recent() already gives chronological slice and the dashboard filters client-side.
            raw = self.memory.recent(limit=limit)
        else:
            raw = self.memory.recent(limit=limit)
    except Exception as exc:  # pragma: no cover — defensive
        log.warning("memory query failed: %s", exc)
        return {"entries": []}
    return {"entries": [self._normalise_memory_entry(r) for r in raw[:limit]]}

async def force_telos(self, params: dict[str, Any]) -> dict[str, Any]:
    """H4 Driver force-Telos. Returns hash diff only — no goal contents cross the RPC boundary.

    params:
        new_telos: dict with key 'active_goals': list[{description, goal_type?, priority?}]
    """
    from noesis_brain.telos.hashing import compute_active_telos_hash
    before = compute_active_telos_hash(self.telos.active_goals())
    new_telos = params.get("new_telos") or {}
    new_goals = new_telos.get("active_goals") if isinstance(new_telos, dict) else None
    if not isinstance(new_goals, list):
        # No-op: hash before == hash after.
        after = compute_active_telos_hash(self.telos.active_goals())
        return {"telos_hash_before": before, "telos_hash_after": after}
    # Rebuild TelosManager from the forced set. Each entry MUST have a description.
    from noesis_brain.telos.manager import TelosManager
    descriptions: list[str] = [str(g.get("description", "")) for g in new_goals if isinstance(g, dict)]
    descriptions = [d for d in descriptions if d]
    self.telos = TelosManager.from_yaml({"short_term": descriptions})
    after = compute_active_telos_hash(self.telos.active_goals())
    return {"telos_hash_before": before, "telos_hash_after": after}
```
