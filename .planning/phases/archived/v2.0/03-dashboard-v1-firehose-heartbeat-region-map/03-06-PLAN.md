    Create `dashboard/src/app/grid/components/region-map.tsx` as a client component (`'use client';`) exporting `RegionMap({ regions, connections }: { regions: Region[]; connections: RegionConnection[] })`.

    Implementation spec:
    ```tsx
    'use client';

    import { memo, useMemo } from 'react';
    import type { Region, RegionConnection } from '@/lib/protocol/region-types';
    import { usePresence } from '../hooks';

    const VIEWPORT_W = 720;
    const VIEWPORT_H = 480;
    const REGION_R = 16;
    const MARKER_R = 5;
    const PADDING = 0.05;         // keeps all coords in [0.05, 0.95]
    const GRID_CELLS = 5;         // 5×5 grid → 25 slots (supports up to ~20 regions before jitter collisions)

    /**
     * FNV-1a 32-bit hash of a string — used for deterministic region placement.
     * We only need stable bucketing; cryptographic properties don\'t matter.
     */
    function hashId(id: string): number {
      let h = 0x811c9dc5;
      for (let i = 0; i < id.length; i++) {
        h ^= id.charCodeAt(i);
        h = Math.imul(h, 0x01000193);
      }
      return h >>> 0; // unsigned 32-bit
    }

    /**
     * Deterministic client-side layout. Resolves Q1b (see 03-RESEARCH.md Open Questions):
     * real Region type has no x/y, so we derive positions from region.id via a stable hash.
     * Exported for test reuse so Playwright can compute expected marker coordinates.
     */
    export function computeRegionLayout(regions: Region[]): Map<string, { x: number; y: number }> {
      const taken = new Set<number>();
      const out = new Map<string, { x: number; y: number }>();
      for (const r of regions) {
        const h = hashId(r.id);
        let cell = h % (GRID_CELLS * GRID_CELLS);
        // Linear-probe on collision so two regions never overlap exactly.
        while (taken.has(cell)) cell = (cell + 1) % (GRID_CELLS * GRID_CELLS);
        taken.add(cell);
        const cx = cell % GRID_CELLS;
        const cy = Math.floor(cell / GRID_CELLS);
        // Center of cell, rescaled into [PADDING, 1 - PADDING].
        const x = PADDING + ((cx + 0.5) / GRID_CELLS) * (1 - 2 * PADDING);
        const y = PADDING + ((cy + 0.5) / GRID_CELLS) * (1 - 2 * PADDING);
        out.set(r.id, { x, y });
      }
      return out;
    }

    export const RegionMap = memo(function RegionMap({
      regions,
      connections,
    }: {
      regions: Region[];
      connections: RegionConnection[];
    }) {
      const presence = usePresence();

      const layout = useMemo(() => computeRegionLayout(regions), [regions]);

      const regionById = useMemo(() => {
        const m = new Map<string, Region>();
        for (const r of regions) m.set(r.id, r);
        return m;
      }, [regions]);

      const project = (regionId: string): { cx: number; cy: number } | null => {
        const pos = layout.get(regionId);
        if (!pos) return null;
        return { cx: pos.x * VIEWPORT_W, cy: pos.y * VIEWPORT_H };
      };

      const edges = useMemo(() => {
        return connections
          .map((c) => {
            const a = project(c.fromRegion);
            const b = project(c.toRegion);
            if (!a || !b) return null;
            return { key: `${c.fromRegion}→${c.toRegion}`, cx: a.cx, cy: a.cy, bx: b.cx, by: b.cy };
          })
          .filter((e): e is NonNullable<typeof e> => e !== null);
      }, [connections, layout]);

      return (
        <svg
          viewBox={`0 0 ${VIEWPORT_W} ${VIEWPORT_H}`}
          role="img"
          aria-label="Region map"
          className="w-full h-full bg-neutral-900 border border-neutral-800 rounded-md"
        >
          {regions.length === 0 && (
            <text x={VIEWPORT_W / 2} y={VIEWPORT_H / 2} textAnchor="middle" className="sr-only">
              No regions loaded
            </text>
          )}

          {/* Edges rendered first so they sit under region circles */}
          <g data-layer="edges">
            {edges.map((e) => (
              <line
                key={e.key}
                data-edge={e.key}
                x1={e.cx}
                y1={e.cy}
                x2={e.bx}
                y2={e.by}
                stroke="#404040"
                strokeWidth={1}
              />
            ))}
          </g>

          {/* Regions */}
          <g data-layer="regions">
            {regions.map((r) => {
              const pos = project(r.id);
              if (!pos) return null;
              return (
                <g
                  key={r.id}
                  data-testid="region-node"
                  data-region-id={r.id}
                >
                  <circle
                    cx={pos.cx}
                    cy={pos.cy}
                    r={REGION_R}
                    fill="#171717"
                    stroke="#525252"
                    strokeWidth={1.5}
                  />
                  <text
                    x={pos.cx}
                    y={pos.cy + REGION_R + 14}
                    textAnchor="middle"
                    fill="#a3a3a3"
                    fontSize={11}
                    fontFamily="Inter, system-ui, sans-serif"
                  >
                    {r.name}
                  </text>
                </g>
              );
            })}
          </g>

          {/* Nous markers — keyed by DID so React preserves nodes across moves */}
          <g data-layer="markers">
            {Array.from(presence.nousByDid.entries()).map(([did, info]) => {
              const pos = project(info.regionId);
              if (!pos) return null;
              return (
                <g
                  key={did}
                  data-testid="nous-marker"
                  data-nous-did={did}
                  style={{
                    transform: `translate(${pos.cx}px, ${pos.cy}px)`,
                    transition: 'transform 150ms ease-out',
                  }}
                >
                  <circle r={MARKER_R} fill="#10b981" />
                  <title>{info.name}</title>
                </g>
              );
            })}
          </g>
        </svg>
      );
    });
    ```

    - `data-testid="region-node"` / `data-testid="nous-marker"` / `data-edge` attributes follow the Plan 02 test-id convention; Playwright smoke and component tests use them as stable selectors.

    export async function startMockGrid(port = 8080) {
      const app = Fastify({ logger: false });
      await app.register(websocketPlugin);

      // CORS parity with Plan 01 — dashboard dev runs at :3001.
      app.addHook('onRequest', async (_req, reply) => {
        reply.header('access-control-allow-origin', 'http://localhost:3001');
      });

      app.get('/api/v1/grid/regions', async () => ({
        regions: REGIONS,
        connections: CONNECTIONS,
      }));

      const sockets = new Set<WebSocket>();

      app.get('/ws/events', { websocket: true }, (conn) => {
        const socket = (conn as unknown as { socket?: WebSocket }).socket ?? (conn as unknown as WebSocket);
        sockets.add(socket);
        const send = (obj: unknown) => socket.send(JSON.stringify(obj));

        socket.on('message', (raw) => {

          let id = 0;
          const emit = (partial: { eventType: string; actorDid: string; payload: Record<string, unknown> }) => {
            id += 1;
