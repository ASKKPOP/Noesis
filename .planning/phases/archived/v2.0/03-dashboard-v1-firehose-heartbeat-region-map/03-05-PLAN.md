       ```typescript
       'use client';
       import { createContext, useContext, useMemo, type ReactNode } from 'react';
       import { FirehoseStore } from '@/lib/stores/firehose-store';
       import { PresenceStore } from '@/lib/stores/presence-store';
       import { HeartbeatStore } from '@/lib/stores/heartbeat-store';

       export interface Stores {
           firehose: FirehoseStore;
           presence: PresenceStore;
           heartbeat: HeartbeatStore;
       }

       const StoresContext = createContext<Stores | null>(null);

       export function StoresProvider({ children }: { children: ReactNode }) {
           const stores = useMemo<Stores>(() => ({
               firehose: new FirehoseStore(),
               presence: new PresenceStore(),
               heartbeat: new HeartbeatStore(),
           }), []);
           return <StoresContext.Provider value={stores}>{children}</StoresContext.Provider>;
       }

       export function useStores(): Stores {
           const ctx = useContext(StoresContext);
           if (!ctx) throw new Error('useStores must be used inside <StoresProvider>');
           return ctx;
       }
       ```
    2. Create `dashboard/src/app/grid/hooks.ts`:
       ```typescript
       'use client';
       import { useSyncExternalStore, useEffect, useState } from 'react';
       import { useStores } from './use-stores';

       export function useFirehose() {
           const { firehose } = useStores();
           return useSyncExternalStore(firehose.subscribe.bind(firehose), firehose.getSnapshot.bind(firehose), firehose.getSnapshot.bind(firehose));
       }

       export function usePresence() {
           const { presence } = useStores();
           return useSyncExternalStore(presence.subscribe.bind(presence), presence.getSnapshot.bind(presence), presence.getSnapshot.bind(presence));
       }

       /**
        * useHeartbeat maintains its own 1s ticker so 'seconds since last event'
        * updates even when no new frames arrive. Returns the DerivedHeartbeat.
        */
       export function useHeartbeat(pollIntervalMs = 1000) {
           const { heartbeat } = useStores();
           const snap = useSyncExternalStore(
               heartbeat.subscribe.bind(heartbeat),
               heartbeat.getSnapshot.bind(heartbeat),
               heartbeat.getSnapshot.bind(heartbeat),
           );
           const [nowMs, setNowMs] = useState(() => Date.now());
           useEffect(() => {
               const id = setInterval(() => setNowMs(Date.now()), pollIntervalMs);
               return () => clearInterval(id);
           }, [pollIntervalMs]);
           return heartbeat.deriveStatus(nowMs);
       }
       ```

       ```tsx
       'use client';
       import { usePresence } from '../hooks';
       import { categorizeEventType } from '@/lib/stores/event-type';
       import type { AuditEntry } from '@/lib/protocol/audit-types';

       const CATEGORY_COLORS: Record<string, string> = {
           trade: 'bg-emerald-500', message: 'bg-sky-500', movement: 'bg-violet-500',
           law: 'bg-amber-500', lifecycle: 'bg-neutral-500', other: 'bg-neutral-700',
       };

       function formatTimestamp(ms: number): string {
           const d = new Date(ms);
           return d.toTimeString().slice(0, 8); // HH:MM:SS
       }

       function payloadPreview(payload: Record<string, unknown>): string {
           const entries = Object.entries(payload).slice(0, 3);
           return entries.map(([k, v]) => `${k}=${typeof v === 'string' ? v.slice(0, 24) : JSON.stringify(v)}`).join(' ');
       }

       export function FirehoseRow({ entry }: { entry: AuditEntry }) {
           const presence = usePresence();
           const name = presence.nameOf(entry.actorDid) ?? entry.actorDid.replace('did:nous:', '').slice(0, 12);
           const category = categorizeEventType(entry.eventType);
           return (
               <li
                   role="listitem"
                   className="h-[28px] flex items-center gap-2 px-3 font-mono text-[12px] border-b border-neutral-800 hover:bg-neutral-900"
                   data-testid="firehose-row"
                   data-event-id={entry.id}
               >
                   <span className="text-neutral-500 w-[72px] shrink-0">{formatTimestamp(entry.createdAt)}</span>
                   <span
                       data-testid="event-type-badge"
                       className={`text-[10px] uppercase tracking-wide px-2 py-0.5 rounded-sm ${CATEGORY_COLORS[category]} text-neutral-950`}
       ```

       ```tsx
       'use client';
       import { useStores } from '../use-stores';
       import { useFirehose } from '../hooks';
       import { ALL_CATEGORIES, type EventCategory } from '@/lib/stores/event-type';

       const DOT: Record<EventCategory, string> = {
           trade: 'bg-emerald-500', message: 'bg-sky-500', movement: 'bg-violet-500',
           law: 'bg-amber-500', lifecycle: 'bg-neutral-500', other: 'bg-neutral-700',
       };

       export function EventTypeFilter() {
           const { firehose } = useStores();
           const snap = useFirehose();
           const current = snap.filter;
           const toggle = (cat: EventCategory) => {
               if (!current) { firehose.setFilter(new Set([cat])); return; }
               const next = new Set(current);
               if (next.has(cat)) next.delete(cat); else next.add(cat);
               firehose.setFilter(next.size === 0 ? null : next);
           };
           return (
               <div className="flex items-center gap-1" role="group" aria-label="Filter by event type">
                   <button
                       type="button"
                       onClick={() => firehose.setFilter(null)}
                       aria-pressed={current === null}
                       className={`text-[11px] px-2 py-0.5 rounded-sm ${current === null ? 'bg-neutral-200 text-neutral-950' : 'bg-neutral-800 text-neutral-300'}`}
                   >All</button>
                   {ALL_CATEGORIES.map(cat => {
                       const on = current?.has(cat) ?? false;
                       return (
                           <button
                               key={cat}
                               type="button"
                               onClick={() => toggle(cat)}
                               aria-pressed={on}
                               className={`flex items-center gap-1 text-[11px] px-2 py-0.5 rounded-sm ${on ? 'bg-neutral-200 text-neutral-950' : 'bg-neutral-800 text-neutral-300'}`}
                           >
                               <span className={`w-1.5 h-1.5 rounded-full ${DOT[cat]}`} />
                               {cat}
                           </button>
                       );
                   })}
               </div>
           );
       }
       ```

       ```tsx
       'use client';
       import { useEffect, useRef } from 'react';
       import { StoresProvider, useStores } from './use-stores';
       import { WsClient } from '@/lib/transport/ws-client';
       import { refillFromDropped } from '@/lib/transport/refill';
       import { Firehose } from './components/firehose';
       import { Heartbeat } from './components/heartbeat';
       import type { Region, RegionConnection } from '@/lib/protocol/region-types';
       import type { AuditEntry } from '@/lib/protocol/audit-types';

       interface GridClientProps {
           origin: string;
           initialRegions: { regions: Region[]; connections: RegionConnection[] } | null;
           initialError: string | null;
       }

       export function GridClient(props: GridClientProps) {
           return (
               <StoresProvider>
                   <GridLayout {...props} />
               </StoresProvider>
           );
       }

       function GridLayout({ origin, initialRegions, initialError }: GridClientProps) {
           const stores = useStores();
           const wsRef = useRef<WsClient | null>(null);

           useEffect(() => {
               const wsUrl = origin.replace(/^http/, 'ws') + '/ws/events';
               const client = new WsClient({ url: wsUrl, onError: console.error });
               wsRef.current = client;

               const abort = new AbortController();
               const ingestAll = (entries: AuditEntry[]) => {
                   stores.firehose.ingest(entries);
                   stores.presence.applyEvents(entries);
                   stores.heartbeat.ingestBatch(entries);
               };

               const offEvent = client.on('event', (entry) => ingestAll([entry]));
               const offDropped = client.on('dropped', async (frame) => {
                   try {
                       const entries = await refillFromDropped(frame, origin, ingestAll, abort.signal);
                       if (entries.length > 0) {
                           const lastId = entries[entries.length - 1].id;
                           if (typeof lastId === 'number') client.bumpLastSeenId(lastId);
                       } else if (frame.latestId > 0) {
                           client.bumpLastSeenId(frame.latestId);
                       }
                   } catch (err) {
                       console.error('refill failed', err);
                   }
               });

               client.connect();

               return () => {
                   offEvent();
                   offDropped();
                   abort.abort();
                   client.close();
                   wsRef.current = null;
               };
           }, [origin, stores]);

           // Region map placeholder (Plan 06 replaces this block)
           const regionMapPlaceholder = (
               <section aria-label="Region map" className="flex items-center justify-center border border-neutral-800 rounded bg-neutral-950 text-neutral-600 text-sm min-h-[320px]">
                   Region map — coming in Plan 06
               </section>
           );

           return (
               <main className="min-h-screen grid grid-cols-1 lg:grid-cols-[1fr_360px] gap-4 p-4 bg-neutral-950">
                   {initialError && (
                       <div role="alert" className="col-span-full text-sm text-amber-400 bg-neutral-900 border border-amber-900 rounded px-3 py-2">
                           Initial regions fetch failed: {initialError}. Reconnect in progress…
                       </div>
                   )}
                   <div className="grid grid-rows-[auto_1fr] gap-4">
                       {regionMapPlaceholder}
                       <Firehose />
                   </div>
                   <aside className="space-y-4">
                       <Heartbeat />
                   </aside>
               </main>
           );
       }
       ```
