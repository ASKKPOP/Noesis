        // When binding 0.0.0.0 behind a reverse proxy, pair GRID_WS_SECRET with an origin
        // allowlist check here. See PITFALLS.md §M5 for the deployment-time checklist.
        //
        // Local-dev auth posture (02-CONTEXT.md §Local-dev auth posture):
        //   - If GRID_WS_SECRET env is set: require Authorization: Bearer <secret>
        //     header OR ?token=<secret> query param. Missing/mismatch → 1008 close.
        //   - If unset: permissive (developer default, 127.0.0.1 bind).
        app.register(async (instance) => {
            instance.get('/ws/events', { websocket: true }, (socket, req) => {
                const secret = process.env.GRID_WS_SECRET;
                if (secret) {
                    const headerAuth = (req.headers['authorization'] as string | undefined) ?? '';
                    const bearer = headerAuth.startsWith('Bearer ')
                        ? headerAuth.substring('Bearer '.length)
                        : null;
                    const queryToken =
                        typeof (req.query as any)?.token === 'string'
                            ? (req.query as any).token as string
                            : null;
                    const presented = bearer ?? queryToken;
                    if (presented !== secret) {
                        try { socket.close(1008, 'unauthorized'); } catch { /* swallow */ }
                        return;
                    }
                }
                // Adapter: fastify/ws socket → ServerSocket shape expected by WsHub.
                const adapter = {
                    get bufferedAmount() { return socket.bufferedAmount; },
                    send: (data: string) => socket.send(data),
                    close: (code?: number, reason?: string) => socket.close(code, reason),
                    on: (event: 'message' | 'close' | 'error', cb: any) => {
                        if (event === 'message') {
                            socket.on('message', (data: Buffer) => cb(data.toString('utf8')));
                        } else if (event === 'close') {
                            socket.on('close', () => cb());
                        } else {
                            socket.on('error', (err: Error) => cb(err));
                        }
                    },
                };
                wsHub.onConnect(adapter, { headers: req.headers as Record<string, unknown> });
            });
        });

        // ── Lifecycle: drain hub on app.close() ──────────────────────────────
        app.addHook('onClose', async () => {
            await wsHub.close();
        });

        return { app, wsHub };
    }

            await waitOpen(a);
            await waitMessage(a); // hello

            const firstIds: number[] = [];
            for (let i = 0; i < 3; i++) {
                const pending = waitMessage(a, 1000);
                audit.append('tick', 'g', { n: i });
                const ev = await pending;
                expect(ev.type).toBe('event');
                firstIds.push(ev.entry.id);
            }
            const lastSeenId = firstIds[firstIds.length - 1];
            expect(lastSeenId).toBeGreaterThan(0);

            // --- Close A, append two "missed" events while disconnected ---
            a.close();
            await new Promise((r) => setTimeout(r, 100));
            expect(wsHub.clientCount).toBe(0);
            audit.append('tick', 'g', { n: 'missed-1' });
            audit.append('tick', 'g', { n: 'missed-2' });

            // --- Phase B: reconnect with sinceId, expect replay ---
            const b = new WsClient(`ws://127.0.0.1:${port}/ws/events`);
            await waitOpen(b);
            await waitMessage(b); // hello

            // Collect all messages B receives; we will classify them.
            const received: any[] = [];
            const collecting = new Promise<void>((resolve) => {
                b.on('message', (buf) => {
                    received.push(JSON.parse(buf.toString()));
                    if (received.length >= 3) resolve();
                });
                setTimeout(() => resolve(), 1500); // safety cutoff
            });

            b.send(JSON.stringify({ type: 'subscribe', sinceId: lastSeenId }));

            // Give the replay a moment, then append one live event.
            await new Promise((r) => setTimeout(r, 100));
            audit.append('tick', 'g', { n: 'live-after-replay' });

            await collecting;

            // Expect: 2 replayed EventFrames (ids = lastSeenId+1, lastSeenId+2)
            //         followed by 1 live EventFrame. NO DroppedFrame.
            const events = received.filter((f) => f.type === 'event');
            const dropped = received.filter((f) => f.type === 'dropped');
            expect(dropped.length).toBe(0);
            expect(events.length).toBeGreaterThanOrEqual(3);

            // The first two events B sees AFTER subscribe must be the two missed ones,
            // in id-ascending order, before the live one.
            const missedPayloads = events.slice(0, 2).map((e) => e.entry.payload.n);
            expect(missedPayloads).toEqual(['missed-1', 'missed-2']);
            // The ids are strictly after lastSeenId.
            expect(events[0].entry.id).toBe(lastSeenId + 1);
            expect(events[1].entry.id).toBe(lastSeenId + 2);
            // The third event (live) arrived after the replay.
            expect(events[2].entry.payload.n).toBe('live-after-replay');
            expect(events[2].entry.id).toBe(lastSeenId + 3);

            b.close();
            await new Promise((r) => setTimeout(r, 50));
        });
    });

## M5 Deferral Summary

Per plan-checker feedback: the M5 pitfall (CORS/Origin on WS upgrade) is consciously deferred for v1 with the following explicit record:

1. **Threat-register entry (T-02-03-09 above)** — classifies the risk as `defer (v1)` with rationale.
2. **Code comment in server.ts** — the literal string `M5:` appears in the `/ws/events` handler and directs future readers to Phase 4 hardening.
3. **Rationale** — default bind is 127.0.0.1; single-developer scope per CONTEXT.md; GRID_WS_SECRET already provides a shared-secret gate when binding 0.0.0.0. A cross-origin browser from a malicious page cannot reach a 127.0.0.1 WS server without the user also exposing the port.
4. **Revisit trigger** — Phase 4 binds 0.0.0.0 behind a reverse proxy; that is when origin allowlist + wss:// termination land together.

This is the auditable trail the plan-checker asked for: an explicit threat-register entry, not an implicit skip.
