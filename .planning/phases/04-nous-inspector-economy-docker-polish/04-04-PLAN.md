    ```typescript
    const DID_REGEX = /^did:noesis:[a-z0-9_\-]+$/i;

    export class SelectionStore {
        private current: string | null = null;
        private readonly listeners = new Set<() => void>();

        subscribe = (listener: () => void): (() => void) => {
            this.listeners.add(listener);
            return () => { this.listeners.delete(listener); };
        };

        getSnapshot = (): string | null => this.current;

        selectNous(did: string | null): void {
            const next = did !== null && DID_REGEX.test(did) ? did : null;
            if (next === this.current) return;
            this.current = next;
            for (const l of this.listeners) l();
        }

        clear(): void { this.selectNous(null); }
    }

    export const selectionStore = new SelectionStore();
    ```

  <action>
    Create `dashboard/src/lib/hooks/use-hash-sync.ts`:
    ```typescript
    'use client';
    import { useEffect, useRef } from 'react';
    import { selectionStore, SelectionStore } from '@/lib/stores/selection-store';

    const NOUS_HASH_RE = /^#nous=(.+)$/;

    function parseHash(hash: string): string | null {
        const m = NOUS_HASH_RE.exec(hash);
        return m ? decodeURIComponent(m[1]) : null;
    }

    function buildHash(did: string | null): string {
        return did ? `#nous=${encodeURIComponent(did)}` : '';
    }

    export function useHashSync(store: SelectionStore = selectionStore): void {
        const lastWritten = useRef<string>('');

        useEffect(() => {
            // Initial read
            const initial = parseHash(window.location.hash);
            if (initial) store.selectNous(initial);

            // Subscribe to store → write URL
            const unsubscribe = store.subscribe(() => {
                const next = buildHash(store.getSnapshot());
                if (window.location.hash === next) return;
                lastWritten.current = next;
                const url = `${window.location.pathname}${window.location.search}${next}`;
                window.history.replaceState(null, '', url);
            });

            // Listen for external hash changes
            const onHashChange = () => {
                if (window.location.hash === lastWritten.current) return;
                const did = parseHash(window.location.hash);
                store.selectNous(did);
            };
            window.addEventListener('hashchange', onHashChange);

            return () => {
                unsubscribe();
                window.removeEventListener('hashchange', onHashChange);
            };
        }, [store]);
    }
    ```

    ```typescript
    'use client';
    import { useRouter, useSearchParams } from 'next/navigation';
    import { useCallback, useRef } from 'react';

    type Tab = 'firehose' | 'economy';
    const TABS: { id: Tab; label: string; testId: string }[] = [
        { id: 'firehose', label: 'Firehose', testId: 'tab-firehose' },
        { id: 'economy',  label: 'Economy',  testId: 'tab-economy'  },
    ];

    export function TabBar() {
        const router = useRouter();
        const searchParams = useSearchParams();
        const active: Tab = searchParams.get('tab') === 'economy' ? 'economy' : 'firehose';
        const refs = useRef<Map<Tab, HTMLButtonElement>>(new Map());

        const activate = useCallback((tab: Tab) => {
            const params = new URLSearchParams(searchParams.toString());
            if (tab === 'economy') params.set('tab', 'economy'); else params.delete('tab');
            router.replace(`?${params.toString()}`);
            refs.current.get(tab)?.focus();
        }, [router, searchParams]);

        const onKeyDown = (e: React.KeyboardEvent<HTMLDivElement>) => {
            if (e.key === 'ArrowRight' || e.key === 'ArrowLeft') {
                const idx = TABS.findIndex(t => t.id === active);
                const delta = e.key === 'ArrowRight' ? 1 : -1;
                const next = TABS[(idx + delta + TABS.length) % TABS.length];
                activate(next.id);
            } else if (e.key === 'Home') activate(TABS[0].id);
            else if (e.key === 'End') activate(TABS[TABS.length - 1].id);
            else return;
            e.preventDefault();
        };

        return (
            <div role="tablist" aria-label="Grid views" data-testid="tab-bar" onKeyDown={onKeyDown}>
                {TABS.map(t => (
                    <button
                        key={t.id}
                        role="tab"
                        aria-selected={active === t.id}
                        tabIndex={active === t.id ? 0 : -1}
                        data-testid={t.testId}
                        ref={(el) => { if (el) refs.current.set(t.id, el); }}
                        onClick={() => activate(t.id)}
                    >{t.label}</button>
                ))}
            </div>
        );
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL hash/search params → SelectionStore | User-controllable input may inject non-DID values or XSS vectors via `#nous=<anything>` |
| SelectionStore → DOM rendering | A selected DID is interpolated into `data-` attributes; must not support HTML escape |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-17 | Tampering | `#nous=` payload injection | mitigate | `DID_REGEX` whitelist in SelectionStore; invalid input normalizes to null, never throws, never propagates unsanitized |
| T-04-18 | Information Disclosure | History entries leaking selections in browser history | accept | Selections are operator UX state, not sensitive; `replaceState` avoids history-stack bloat |
| T-04-19 | Denial of Service | Infinite loop between store update → hash write → hashchange → store update | mitigate | `lastWritten` ref + `===` guard in useHashSync; explicitly tested |
| T-04-20 | Spoofing | Cross-tab URL coordination confusion | accept | Per-tab store instance; no cross-tab broadcast by design |
</threat_model>
