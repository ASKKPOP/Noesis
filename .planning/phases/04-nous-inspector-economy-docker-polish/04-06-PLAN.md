    ```typescript
    export type NousRosterEntry = {
        did: string; name: string; region: string; ousia: number;
        lifecyclePhase: string; reputation: number; status: string;
    };
    export type RosterResponse = { nous: NousRosterEntry[] };

    export type TradeRecord = {
        actorDid: string; counterparty: string; amount: number;
        nonce: string;
        timestamp: number; // Unix SECONDS (integer) — W2 contract locked in Plan 04-03
    };
    export type TradesResponse = { trades: TradeRecord[]; total: number };

    export type ShopListing = { sku: string; label: string; priceOusia: number };
    export type Shop = { ownerDid: string; name: string; listings: ShopListing[] };
    export type ShopsResponse = { shops: Shop[] };

    type Result<T> = { ok: true; data: T } | { ok: false; error: { kind: 'network' } };

    async function getJson<T>(url: string, signal?: AbortSignal): Promise<Result<T>> {
        let resp: Response;
        try {
            resp = await fetch(url, { signal, headers: { accept: 'application/json' } });
        } catch (err) {
            if ((err as { name?: string }).name === 'AbortError') throw err;
            return { ok: false, error: { kind: 'network' } };
        }
        if (!resp.ok) return { ok: false, error: { kind: 'network' } };
        return { ok: true, data: (await resp.json()) as T };
    }

    export function fetchRoster(origin: string, signal?: AbortSignal) {
        return getJson<RosterResponse>(`${origin}/api/v1/grid/nous`, signal);
    }

    export function fetchTrades(origin: string, signal?: AbortSignal, limit = 20) {
        const n = Number.isFinite(limit) && limit >= 1 ? Math.min(limit, 100) : 20;
        return getJson<TradesResponse>(`${origin}/api/v1/economy/trades?limit=${n}`, signal);
    }

    export function fetchShops(origin: string, signal?: AbortSignal) {
        return getJson<ShopsResponse>(`${origin}/api/v1/economy/shops`, signal);
    }
    ```

    ```typescript
    'use client';
    import { useEffect, useMemo, useRef, useState, useSyncExternalStore } from 'react';
    import { fetchRoster, fetchTrades, fetchShops, type RosterResponse, type TradesResponse, type ShopsResponse } from '@/lib/api/economy';
    import { useStores } from '../use-stores';
    import { BalancesTable } from './balances-table';
    import { TradesTable } from './trades-table';
    import { ShopsList } from './shops-list';
    import { EmptyState } from '@/components/primitives';

    type Slot<T> = { loading: true } | { loading: false; data: T } | { loading: false; error: true };

    export function EconomyPanel() {
        const stores = useStores();
        const events = useSyncExternalStore(stores.firehose.subscribe, stores.firehose.getSnapshot, stores.firehose.getSnapshot);
        const origin = useMemo(() => process.env.NEXT_PUBLIC_GRID_ORIGIN ?? '', []);
        const [roster, setRoster] = useState<Slot<RosterResponse>>({ loading: true });
        const [trades, setTrades] = useState<Slot<TradesResponse>>({ loading: true });
        const [shops,  setShops ] = useState<Slot<ShopsResponse>>({ loading: true });
        const lastTradeId = useRef<string | number | null>(null);

        const loadRosterAndTrades = (ac: AbortController) => Promise.all([
            fetchRoster(origin, ac.signal).then(r => setRoster(r.ok ? { loading: false, data: r.data } : { loading: false, error: true })),
            fetchTrades(origin, ac.signal).then(r => setTrades(r.ok ? { loading: false, data: r.data } : { loading: false, error: true })),
        ]);

        useEffect(() => {
            const ac = new AbortController();
            loadRosterAndTrades(ac);
            fetchShops(origin, ac.signal).then(r => setShops(r.ok ? { loading: false, data: r.data } : { loading: false, error: true }));
            return () => ac.abort();
            // eslint-disable-next-line react-hooks/exhaustive-deps
        }, [origin]);

        // Firehose-driven invalidation for trade.settled
        useEffect(() => {
            const latest = [...events].reverse().find(e => (e as { eventType?: string }).eventType === 'trade.settled');
            if (!latest) return;
            const id = (latest as { id?: string | number }).id ?? null;
            if (id === null || id === lastTradeId.current) return;
            lastTradeId.current = id;
            const ac = new AbortController();
            loadRosterAndTrades(ac);
            return () => ac.abort();
            // eslint-disable-next-line react-hooks/exhaustive-deps
        }, [events, origin]);

        return (
            <div data-testid="economy-panel">
                <h2 className="mb-2 text-sm font-semibold text-slate-100">Economy</h2>
                {roster.loading ? <div data-testid="balances-loading">Loading balances…</div>
    ```
